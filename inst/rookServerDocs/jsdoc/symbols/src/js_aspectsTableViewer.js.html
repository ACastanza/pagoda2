  8 /**
  9  * Manages the aspects table
 10  * @constructor
 11  */
 12 function aspectsTableViewer() {
 13   if (typeof aspectsTableViewer.instance === 'object'){
 14     return aspectsTableViewer.instance;
 15   }
 16 
 17   console.log('Initializing aspects table viewer...');
 18 
 19   // Generate the extjs elemetns
 20   this.generateTables();
 21 
 22   // Load the data
 23   // NOTE: There is a race condition here between the generation of the tables
 24   // and the ajax call
 25   var dataCntr = new dataController();
 26   dataCntr.getAvailableAspectsStore(function(aspectTableStore){
 27         var table = Ext.getCmp('aspectstable');
 28         table.bindStore(aspectTableStore);
 29   });
 30 
 31   aspectsTableViewer.instance = this;
 32 };
 33 
 34 aspectsTableViewer.prototype.updateGeneSetsInAspectTable = function(selectedAspect) {
 35       var dataCntr = new dataController();
 36       dataCntr.getAvailableGenesetsInAspectStore(selectedAspect, function(store) {
 37           var table = Ext.getCmp('genesetsAspectTable');
 38           table.bindStore(store);
 39 
 40 
 41       });
 42 };
 43 
 44 aspectsTableViewer.prototype.generateTables = function() {
 45   var dataCntr = new dataController() ;
 46 
 47   var areaHolder = Ext.getCmp('aspectsExtJS');
 48 
 49   //
 50 
 51   // Selection change listener for aspects
 52   var aspectSelectionChangeListener = function(obj, selected, eOpts) {
 53     var selectedAspect = selected[0].data.name;
 54     var aspTableViewer = new aspectsTableViewer();
 55     aspTableViewer.updateGeneSetsInAspectTable(selectedAspect);
 56   };
 57 
 58   var aspectSelectionGrid = Ext.create('Ext.grid.Panel',{
 59     title: 'Available Aspects',
 60     id: 'aspectstable',
 61     empty: 'No aspects are defined',
 62     columns: [{text:'Name', dataIndex:'name', width:'100%'}],
 63     listeners: {
 64       'selectionchange': aspectSelectionChangeListener
 65     }
 66   });
 67 
 68   var genesetSelectionChangeListener = function(obj, selected, eOpts) {
 69     // FIXME: Here we assume that they are all go terms
 70     // and they are also provided in the genesets of interest
 71 
 72     var selectedGeneset = selected[0].data.name;
 73     var goTerm = selectedGeneset.replace(/#.*# /,'');
 74 
 75     var dataCntr = new dataController();
 76     dataCntr.getGeneSetStoreByName(goTerm, function(store) {
 77       var genesetTable = Ext.getCmp('genesAspectsTable');
 78 		  store.sort({property: 'dispersion', direction: 'DESC'});
 79 		  genesetTable.bindStore(store);
 80 
 81     })
 82 
 83   }
 84 
 85   var setsTableForAspects = Ext.create('Ext.grid.Panel',{
 86     title: 'Gene sets in Aspect',
 87     id: 'genesetsAspectTable',
 88     empty: 'No sets available',
 89     columns: [{text:'Name', dataIndex:'name', width:'100%'}],
 90     listeners: {
 91       'selectionchange': genesetSelectionChangeListener
 92     }
 93   });
 94 
 95   var selModel =  Ext.create('Ext.selection.CheckboxModel', {});
 96 
 97   var genesTableForAspects = Ext.create('Ext.grid.Panel',{
 98     title: 'Genes in the selected Gene Set',
 99     id: 'genesAspectsTable',
100     empty: 'No genes selected',
101     selModel: selModel,
102     columns: [
103       {
104         text: 'Name', dataIndex: 'genename', width: '80%',
105         renderer: function(value) {return Ext.String.format(p2globalParams.misc.jaxGeneQueryFormatString ,value,value)}
106       },
107       {
108         text: 'Dispersion',
109         dataIndex: 'dispersion',
110         width: '20%'
111       }], //columns
112       height: '100%',
113       width: '100%',
114       singleSelect: false,
115       listeners: {
116         'selectionchange': function(selected, eOpts) {
117           		var selectedGeneNames =  [];
118 
119       		    var selectedItems = selected.getSelected();
120       		    selectedItems.each(function(item,index,length){
121       			    selectedGeneNames.push(item.data.genename);
122       		    });
123 
124       		    var geneSelCntr =  new geneSelectionController();
125       		    geneSelCntr.setSelection('geneTableSelection', selectedGeneNames);
126         } // selection change
127       }, // listeners
128       tbar: Ext.create('Ext.PagingToolbar', {
129       		displayInfo: true,
130 		      prependButtons: true,
131 		      items: [
132             {
133       			type: 'button',
134       			text: 'Show selected',
135       			tooltip: 'Show selected genes in the main heatmap',
136       			glyph: 0xf0ce,
137       			handler: function() {
138       			    var heatmapV = new heatmapViewer();
139       			    heatmapV.setNamedSelectionToDisplayGenes('geneTableSelection');
140       			    heatmapV.drawHeatmap();
141       			} //handler
142           }, //item
143          ] //items
144       }) //tbar
145   });
146 
147   areaHolder.add([{
148     type: 'panel',
149     title: '',
150     header: false,
151     region: 'north',
152     split: true,
153     layout: 'fit',
154     width: '100%',
155     height: '25%',
156     items: [aspectSelectionGrid]
157   },{
158     type: 'panel',
159     title: '',
160     layout: 'fit',
161     header: false,
162     region: 'center',
163     split: true,
164     width: '100%',
165     height: '25%',
166     items: [setsTableForAspects]
167   },{
168     type: 'panel',
169     title: '',
170     layout: 'fit',
171     header: false,
172     region: 'south',
173     split: true,
174     width: '100%',
175     height: '50%',
176     items: [genesTableForAspects]
177   }]);
178 };

180 /**
181  * Responds to an external aspect selection
182  * @description responds to an aspect selection, it brings up the aspects tag
183  * and selects the designated aspect
184  */
185 aspectsTableViewer.prototype.showSelectedAspect = function(aspectIdentifier) {
186   this.raiseTab();
187 
188   var table = Ext.getCmp('aspectstable');
189   var index = table.getStore().find('name', aspectIdentifier);
190   table.getSelectionModel().select(index);
191 };
192 
193 /**
194  * Raised the tab to of this panel
195  */
196 aspectsTableViewer.prototype.raiseTab = function() {
197   var tablesTab = Ext.getCmp('tablesTabExtJS');
198   // FIXME: The tab order is hard-wired here
199   tablesTab.setActiveTab(2);
200 };
