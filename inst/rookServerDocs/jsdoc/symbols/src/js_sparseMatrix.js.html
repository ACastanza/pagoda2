  8 /**
  9  * A read-only compressed sparse row matrix
 10  * @constructor
 11  */
 12 function dgCMatrixReader(i, p, Dim, Dimnames1, Dimnames2, x, factors) {
 13     this.i =  i;
 14     this.p =  p;
 15     this.Dim = Dim;
 16     this.DimNames1 = Dimnames1;
 17     this.DimNames2 = Dimnames2;
 18     this.x = x;
 19     this.factors = factors;
 20 }
 21 
 22 /**
 23  * Get a full js matrix out of the sparse matrix representation
 24  */
 25 dgCMatrixReader.prototype.getFullMatrix = function() {   
 26 
 27     // Make a zero filled array
 28     var out = Array(this.Dim[0]);
 29     for (var k = 0; k < this.Dim[0]; k++) {
 30 	var row = Array(this.Dim[1]);
 31 	for (var j = 0; j < this.Dim[1]; j++) {
 32 	    row[j] = 0;
 33 	}
 34 	out[k] = row;
 35     }
 36 
 37 
 38     // index in p (the column number)
 39     for (var j = 0; j < this.p.length - 1; j++) {
 40 	// row start index, row end index (in x and i)
 41 	var rsi = this.p[j];
 42 	var rei = this.p[j+1] - 1;
 43 
 44 	// k is an index in x and i with the current column
 45 	for (var k = rsi; k < rei; k++) {
 46 	    // row number
 47 	    var rn = this.i[k];
 48 	    
 49 	    // x[k] is the value for the element in rn, j
 50 	    out[rn][j] = this.x[k];
 51 	}
 52     }
 53 
 54     var retVal = {
 55 	array:  out,
 56 	rownames:  this.DimNames1,
 57 	colnames:  this.DimNames2
 58     };
 59 
 60     return retVal;
 61 }
