  8 /**
  9  * Plots the scatter plot using canvas
 10  * @constructor
 11  */
 12 function embeddingViewerScatterCanvas() {
 13     console.log('Initializing scatterEmbeddingCanvas...');
 14 
 15     // Here we need to tap into the extjs container
 16     // component resize function 
 17     var element = Ext.getCmp('embedding-app-container');
 18     element.onResize = function(){
 19 	embView = new embeddingViewer();
 20 	embView.redraw();
 21     };
 22 
 23     var evtBus = new eventBus();
 24     // Dendrogram node has been clicked
 25     // Set the coloring scheme to dendrogram ( which uses the selection controller )
 26     // and call updateColors
 27     evtBus.register("dendrogram-node-click", null, function(e,p) {
 28 	var embV = new embeddingViewer();
 29 
 30 	// Update  configuration of embedding
 31 	// The actual selected node is passed by the cell selection
 32 	embV.setColorConfiguration('dendrogram');
 33 	embV.setDendrogramColorInfo({nodeType: p.type});
 34 
 35 	embV.updateColors();
 36     }); 
 37 
 38     // This variable hold the last ajax request for expression values
 39     // The purpose of this is to be able to cancel this request
 40     // If the user makes a new request while the current one is running
 41     this.expressionValuesAJAXrequest;
 42 
 43     // We use this to cache the cell order as calculated by getExpression
 44     // and avoid recalculating every time. We need to cache this for 
 45     // each embedding seperately
 46     this.fillStylesGeneExpressionOrder = {};
 47 
 48     // Used to slightly shrink the plotting area to prevent
 49     // plotting on the edges
 50     this.rangeScaleFactor = 0.95;
 51 
 52     // These are used to keep track of the 
 53     // Transformation applied to the embedding points
 54     // This is needed so that selections can be made 
 55     this.xScaleDomainMin;
 56     this.xScaleDomainMax;
 57     this.xScaleRangeMin;
 58     this.xScaleRangeMax;
 59 
 60     this.yScaleDomainMin;
 61     this.yScaleDomainMax;
 62     this.yScaleRangeMin;
 63     this.yScaleRangeMax;
 64 
 65     // Generates the html structure required for the viewer
 66     this.generateStructure();    
 67 }

 70 /**
 71  * Generate the selection following the completion of the 
 72  * drag event.
 73  * @description Perform the actions necessary at the end of  
 74  * the drag event completion. Specifically, clear the overlay
 75  * plot only the selected points onto the overlay
 76  * and set the cell selection
 77  */
 78 embeddingViewerScatterCanvas.prototype.generateDragSelection = 
 79     function(dragStartX, dragStartY, dragEndX, dragEndY) {
 80 	if (dragStartX > dragEndX) {
 81 	    var t = dragEndX;
 82 	    dragEndX =  dragStartX;
 83 	    dragStartX = t;
 84 	}
 85 	if (dragStartY > dragEndY) {
 86 	    var t = dragEndY;
 87 	    dragEndY = dragStartY;
 88 	    dragStartY = t;
 89 	}
 90 	
 91 
 92 	thisViewer = this;
 93 	var embViewer = new embeddingViewer();
 94 	config = embViewer.getConfig();
 95 
 96 	// Reversed scales
 97 	var xScaleRev = pagHelpers.linearScaleGenerator(
 98 	    thisViewer.xScaleRangeMin,
 99 	    thisViewer.xScaleRangeMax,
100 	    thisViewer.xScaleDomainMin,
101 	    thisViewer.xScaleDomainMax);
102 
103 	var yScaleRev = pagHelpers.linearScaleGenerator(
104 	    thisViewer.yScaleRangeMin,
105 	    thisViewer.yScaleRangeMax,
106 	    thisViewer.yScaleDomainMin,
107 	    thisViewer.yScaleDomainMax);
108 
109 	var x1 = xScaleRev(dragStartX);
110 	var y1 = yScaleRev(dragStartY);
111 	var x2 = xScaleRev(dragEndX);
112 	var y2 = yScaleRev(dragEndY);
113 
114 	var dataCntr = new dataController();
115 	var type = config.type;
116 	var embeddingType = config.embeddingType;
117 
118 	dataCntr.getEmbedding(type, embeddingType, function(data) {
119 	    plotData = pagHelpers.jsonSerialisedToArrayOfArrays(data);
120 	    
121 	    
122 	    // Make the xscale
123 	    thisViewer.xScaleDomainMin = +Infinity;
124 	    thisViewer.xScaleDomainMax = -Infinity;
125 	    for (j = 0; j < plotData.length; j++) {
126 		thisViewer.xScaleDomainMin = Math.min(thisViewer.xScaleDomainMin, plotData[j][1]);
127 		thisViewer.xScaleDomainMax = Math.max(thisViewer.xScaleDomainMax, plotData[j][1]);
128 	    }
129 	    thisViewer.xScaleRangeMin = (size * (1 - thisViewer.rangeScaleFactor));
130 	    thisViewer.xScaleRangeMax = size * thisViewer.rangeScaleFactor;
131 	    var xScale = pagHelpers.linearScaleGenerator(thisViewer.xScaleDomainMin, 
132 							 thisViewer.xScaleDomainMax, 
133 							 thisViewer.xScaleRangeMin, 
134 							 thisViewer.xScaleRangeMax);
135 
136 	    // Make the y scale
137 	    thisViewer.yScaleDomainMin = +Infinity;
138 	    thisViewer.yScaleDomainMax = -Infinity;
139 	    for (j = 0; j < plotData.length; j++) {
140 		thisViewer.yScaleDomainMin = Math.min(thisViewer.yScaleDomainMin, plotData[j][2]);
141 		thisViewer.yScaleDomainMax = Math.max(thisViewer.yScaleDomainMax, plotData[j][2]);
142 	    }
143 	    thisViewer.yScaleRangeMin = (size * (1 - thisViewer.rangeScaleFactor));
144 	    thisViewer.yScaleRangeMax = size * thisViewer.rangeScaleFactor;
145 	    var yScale = pagHelpers.linearScaleGenerator(thisViewer.yScaleDomainMin,
146 							 thisViewer.yScaleDomainMax, 
147 							 thisViewer.yScaleRangeMin,
148 							 thisViewer.yScaleRangeMax);
149 
150 	    var pointsize = embViewer.getCurrentPointSize();
151 
152 	    var ctx = document.getElementById('embedding-canvas-overlay').getContext('2d');
153 	    ctx.clearRect(0,0,5000,5000);
154 
155 	    var cellsForSelection = new Array();
156 	    ctx.strokeStyle = 'red';
157 
158 	    for (var i = 0; i < plotData.length; i++) {
159 		var point = plotData[i];
160 		
161 		if ( point[1] > x1 && point[1] < x2 && point[2] > y1 && point[2] < y2) {
162 		    // Point in selection
163 		    cellsForSelection.push(point[0]);
164 		    
165 		    var xs = xScale(point[1]);
166 		    var ys = yScale(point[2]);
167 		    
168 		    ctx.beginPath();
169 		    ctx.arc(xs, ys, pointsize, 0, 2 * Math.PI, false);
170 		    ctx.stroke();
171 		}
172 	    } // for
173 
174 	    var cellSelCntr = new cellSelectionController();
175 	    cellSelCntr.setSelection('embSelection', cellsForSelection, 'Embedding Selection', new Object());
176 
177 	});
178 
179 
180     }

210 embeddingViewerScatterCanvas.prototype.setupOverlayEvents = function(overlayCanvasElement) {
211     var thisViewer = this;
212 
213     var dragging = false;
214     var dragStartX = 0;
215     var dragStartY = 0;
216     var ctx = overlayCanvasElement.getContext('2d');
217 
218     overlayCanvasElement.addEventListener('mouseover', function(e) {
219 	document.body.style.cursor = 'crosshair';
220     });
221 
222     overlayCanvasElement.addEventListener('mousedown', function(e) {
223 	dragStartX = e.layerX;
224 	dragStartY = e.layerY;
225 	dragging = true;
226     });
227 
228     overlayCanvasElement.addEventListener('mouseup', function(e) {
229 	if (dragging) {
230 	    // Dragging complete
231 	    var dragEndX = e.layerX;
232 	    var dragEndY = e.layerY;
233 
234 	    
235 
236 	    thisViewer.generateDragSelection(dragStartX, dragStartY, dragEndX, dragEndY);
237 	}
238 	dragging = false;
239     });
240     
241     overlayCanvasElement.addEventListener('mousemove', function(e) {
242 	var dragEndX =  e.layerX;
243 	var dragEndY = e.layerY;
244 	
245 
246 	if (dragging) {
247 	    // TODO: clear the correct coordinates
248 	    ctx.clearRect(0,0,4000,4000);
249 	    ctx.save();
250 	    ctx.setLineDash([10,10]);
251 	    ctx.strokeStyle = '#FF0000';
252 	    ctx.lineWidth = 2;
253 	    ctx.strokeRect(dragStartX, dragStartY, dragEndX - dragStartX, dragEndY - dragStartY);
254 	    ctx.restore();
255 	}
256 
257     });
258 
259     overlayCanvasElement.addEventListener('mouseleave', function(e) {
260 
261 	if (dragging) {
262 	    ctx.clearRect(0,0,4000,4000);
263 	}
264 	dragging = false;
265 	document.body.style.cursor = 'default';
266     });
267

390 							 thisViewer.xScaleDomainMax, 
391 							 thisViewer.xScaleRangeMin,

404 							 thisViewer.yScaleDomainMax, 
405 							 thisViewer.yScaleRangeMin,
406 							 thisViewer.yScaleRangeMax);
407 
408 	    // The size to plot at
409 	    var pointsize = embViewer.getCurrentPointSize();
410 	    
411 	    // The border
412 	    var stroke;
413 	    var strokeColor;
414 	    var strokeWidth = 1;
415

417

426 	    t0 = performance.now()
427 	    // Main plot loop
428 	    for (var i = 0; i < plotData.length; i++) {
429 		var point = plotData[i];
430 
431 		var xs = xScale(point[1]);
432 		var ys = yScale(point[2]);

440

443 		    ctx.lineWidth = strokeWidth;
444 		    ctx.strokeStyle = strokeColor;
445 		    ctx.stroke();
446 		}
447 	    }
448 
449 	    embViewer.hideWait();
450 
451 
452 	}, type, embeddingType); // GenerateFillStyles
453

460 /** 
461  * Generate fill styles suitable for the given dataset
462  */
463 embeddingViewerScatterCanvas.prototype.generateFillStyles = function(plotdata, callback, type, embeddingType) {
464 
465 
466 
467     // Get the color config
468     var ev = new embeddingViewer();
469     var config = ev.getConfig();
470     var colconfig = config.colors;
471 
472     if (colconfig === 'dendrogram') {
473 	this.generateFillStylesDendrogram(plotdata, callback);
474     } else if (colconfig === 'metadata') {
475 	this.generateFillStylesMetadata(plotdata, callback);
476     } else if (colconfig === 'geneexpression')  {
477 	// Type and embedding type is passed here for caching the matches correctly
478 	this.generateFillStylesGeneExpression(plotdata, callback, type, embeddingType);
479     } else {
480 	this.generateDefaultFill(plotdata,callback);
481     }
482 }

497 /**
498  * Update the alpha values of an array of colors
499  * @colors array of colors in #FFFFFF format
500  * @newAlpha new alpha value to set in the range of 0 to 1
501  * @return array of the colors with alpha set in rgba(1,1,1,1) text format
502  */
503 embeddingViewerScatterCanvas.prototype.updateColorAlpha = function(colors, newAlpha) {
504     // Prepare the palette with our alpha
505     var retVals = [];
506     for (var k = 0; k < colors.length; k++) {
507 	color = colors[k];
508 	
509 	// Split the channels
510 	var r = parseInt(color.substring(1,3),16);
511 	var g = parseInt(color.substring(3,5),16)
512 	var b = parseInt(color.substring(5,7),16);
513 
514 	// Add user-specified alpha
515 	color = 'rgba('+ r+','+g+','+b+','+newAlpha+')';
516 	retVals[k] = color;
517     }
518     return retVals;
519 
520 }

522 /** 
523  * Annotate the plotdata with color from the dendrogram selections and 
524  * call the callback with it.
525  */
526 embeddingViewerScatterCanvas.prototype.generateFillStylesDendrogram = function (plotdata, callback) {
527     var embV = new embeddingViewer();
528     var selCntr =  new cellSelectionController();
529 
530     // Out plot colors in rgb
531     var selectedRGB = p2globalParams.dendrogram.selectionColors.selectedMainRGB;
532     var selectedAltRGB =  p2globalParams.dendrogram.selectionColors.selectedAltRGB;
533     var deselectedRGB =  p2globalParams.dendrogram.selectionColors.deselectedRGB;
534     
535     var alpha = embV.getCurrentAlpha();
536 
537     if (embV.currentConfiguration.dendrogramColorInfo.nodeType == "vertical") {
538 	var curPrimSel = selCntr.getSelection("currentPrimarySel");
539 	for (var i = 0; i < plotdata.length; i++) {
540 	    var cellid = plotdata[i][0];
541 	    if ($.inArray(cellid, curPrimSel) > -1) { 
542 		plotdata[i][3] = 'rgba(' +  selectedRGB[0] + ',' +  selectedRGB[1] + 
543 		    ' , ' + selectedRGB[2] +  ','  + alpha  + ')';
544 	    } else {
545 		plotdata[i][3] =  'rgba(' +  deselectedRGB[0] + ',' +  deselectedRGB[1] + 
546 		    ' , ' + deselectedRGB[2] +  ','  + alpha  + ')';
547 	    }		 
548 	}
549 	callback(plotdata);
550 	return;
551     } else if (embV.currentConfiguration.dendrogramColorInfo.nodeType == "horizontal") {
552 	var curPrimSel = selCntr.getSelection("currentPrimarySel");
553 	var curAltSel = selCntr.getSelection("currentAltSel");
554 	for (var i = 0; i < plotdata.length; i++) {
555 	    var cellid = plotdata[i][0];
556 	    if ($.inArray(cellid, curPrimSel) > -1) {
557 		plotdata[i][3] = 'rgba(' +  selectedRGB[0] + ',' +  selectedRGB[1] + 
558 		    ' , ' + selectedRGB[2] +  ','  + alpha  + ')';
559 	    } else if ($.inArray(cellid, curAltSel) > -1) {
560 		plotdata[i][3] = 'rgba(' +  selectedAltRGB[0] + ',' +  selectedAltRGB[1] + 
561 		    ' , ' + selectedAltRGB[2] +  ','  + alpha  + ')';
562 	    } else {
563 		plotdata[i][3] =  'rgba(' +  deselectedRGB[0] + ',' +  deselectedRGB[1] + 
564 		    ' , ' + deselectedRGB[2] +  ','  + alpha  + ')';
565 	    }    
566 	}
567 	callback(plotdata);
568 	return;
569     }
570 }

573 /**
574  * Generate fill style information from metadata
575  * @param plotdata the plotdata array to augment with colors
576  * @callback the callback function
577  */
578 embeddingViewerScatterCanvas.prototype.generateFillStylesMetadata = function(plotdata, callback) {
579     // Get the configuration information that we need
580     var ev = new embeddingViewer();
581     var config = ev.getConfig();
582 
583     var dataCntr = new dataController();
584 
585     var alpha = ev.getCurrentAlpha();
586 
587     // Get the embedding viewer config and the name
588     // of the metadata we want to plot
589     var metadataColorInfo = config.metaDataColorInfo;
590     var metadataName = metadataColorInfo.metadataName;
591 
592     dataCntr.getCellMetadata(function(metadata) {
593 	colorData = metadata[metadataName].data;
594 	colorPalette = metadata[metadataName].palette;
595 
596 	for (var i = 0; i < plotdata.length; i++) {
597 	    var cellId = plotdata[i][0];
598 	    var clusterId = colorData[cellId];
599 	    var clusterColor;
600 	    
601 	    if (typeof clusterId === 'undefined') {
602 		// There is a cell that is not in the metadata
603 		console.warn('Embedding plotter found a cell that does not have and entry ' +
604 			     'in the selected metadata. Cellid: "' + cellId + '" metadata: "' +
605 			     metadataName + '". This is an error with the data provided by server.');
606 		clusterColor = '#000000'; // default to black
607 	    } else {
608 		clusterColor = colorPalette[clusterId-1];
609 	    }
610 
611 	    if (typeof clusterColor == 'undefined') {
612 		debugger;
613 	    }
614 
615 	    // TODO use objct function here
616 	    // Split the channels
617 	    var r = parseInt(clusterColor.substring(1,3),16);
618 	    var g = parseInt(clusterColor.substring(3,5),16)
619 	    var b = parseInt(clusterColor.substring(5,7),16);
620 
621 	    // Add user-specified alpha
622 	    clusterColor = 'rgba('+ r+','+g+','+b+','+alpha+')';
623 
624 	    plotdata[i][3] = clusterColor;
625 	}
626 	callback(plotdata);
627 	return;
628     });
629

632 /**
633  * Generate fill styles by gene expression
634  * @param plotdata the plotdata to augment with colors
635  * @param callback the callback function
636  * @param type the embedding type, used for caching the match indexes
637  * @param embeddingType the embedding type, used for caching the match indexes
638  */
639 embeddingViewerScatterCanvas.prototype.generateFillStylesGeneExpression = function(plotdata, callback, type, embeddingType) {
640 
641 
642 
643     var evSC = this;
644 
645     var ev = embeddingViewer();   
646     var config = ev.getConfig();
647     var dataCntr = new dataController();
648     var alpha = ev.getCurrentAlpha();
649 
650     // Check if there is a request running already
651     if (typeof evSC.expressionValuesAJAXrequest !== 'undefined') {
652 
653 	if (evSC.expressionValuesAJAXrequest.readyState != 4) {  //Not DONE
654 	    // A previous request exists and is still running
655 	    // cancel it
656 	    evSC.expressionValuesAJAXrequest.abort();
657 	}
658     }
659     
660     // We only end up using this for the number of cell 
661     // which is a complete waste, but it takes less than 1ms so not worth removing
662     dataCntr.getCellOrder (function(data) {
663 	var cellIndexStart = 0;
664 	var cellIndexEnd = data.length;
665 	var geneSelection = [config.geneexpressionColorInfo.geneid];
666 
667 	evSC.expressionValuesAJAXrequest = 
668 	    dataCntr.getExpressionValuesSparseByCellIndex(
