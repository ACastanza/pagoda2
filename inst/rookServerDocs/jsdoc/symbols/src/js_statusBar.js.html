  2 /**
  3  * Responsible for the status bar display item.
  4  * @description On initialisation it registers various event handlers
  5  * to display things. Other object can directly call its showMessage()
  6  * function to display items.
  7  * @constructor
  8  * @todo Minimum time that the displayed item will stay on screen 
  9  * @todo implement icons (i.e. cell, warning, error, ...)
 10  */
 11 function statusBar() {
 12     if (typeof statusBar.instance === 'object') {
 13 	return statusBar.instance;
 14     };
 15 
 16     console.log('Initialising status bar...');
 17 
 18     // Respond to various events
 19     var evtBus = new eventBus();
 20     evtBus.register("embedding-mouseover-plotpoint", null, function(e,p){
 21 
 22 	var stsBar =  new statusBar();
 23 	stsBar.showMessage("Cell Id: " + p.cellId);
 24     });
 25 
 26     statusBar.instance = this;
 27 };
 28 
 29 /**
 30  * Update the message in the status bar
 31  */
 32 statusBar.prototype.showMessage =  function(msg) {
 33 	var extJsStatusBar = Ext.getCmp('pagoda-status-bar');
 34 	extJsStatusBar.setStatus({
 35 	    text: msg
 36 	});
 37 };
