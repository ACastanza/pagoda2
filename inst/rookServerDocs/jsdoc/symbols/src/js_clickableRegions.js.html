 38 /**
 39  * resolve clicks manually by iterating throught the click
 40  * area array, as opposed to using the cache. Called by resolveClick()
 41  * @private
 42  * @param x the x coordinate
 43  * @param y the y coordinate
 44  * @param callback optional callback if not defined the data is returned
 45  * @return the data associated with the region at the time it was created. Returned
 46  * only if a callback is not specified
 47  */
 48 clickableRegions.prototype.resolveClickManual =  function (x,y, callback) {
 49     for (var i = 0; i < this.clickAreas.length; i++) {
 50 	var vs = this.clickAreas[i].vertices
 51 	if (pointInPolygon([x,y], vs)) {
 52 	    if ( typeof callback !== 'undefined' ) {
 53 		callback(this.clickAreas[i].data);
 54 	    } else {
 55 		return this.clickAreas[i].data;
 56 	    }
 57 	    break;
 58 	}; // if
 59     }; // for
 60 }

 91 /**
 92  * Add a new click area bounded by four points
 93  * @description: It is important that click area addition is done through

 96  * @param x1 x coordinate of point 1
 97  * @param y1 y coordinate of point 1
 98  * @param x2 x coordinate of point 2
 99  * @param y2 y coordinate of point 2
100  * @param x3 x coordinate of point 3
101  * @param y3 y coordinate of point 3
102  * @param x4 x coordinate of point 4
103  * @param y4 y coordinate of point 4
104  * @param data data associated with this click area that will be passed to the callback on resolution
105  */
106 clickableRegions.prototype.addClickArea = function(x1, y1, x2, y2, x3, y3, x4, y4, data) {
107     var clkArea = new Object();
108     clkArea["vertices"]  = [ [x1, y1],[x2, y2],[x3, y3],[x4, y4] ];

110 
111     this.clickAreas[this.clickAreas.length] = clkArea;
