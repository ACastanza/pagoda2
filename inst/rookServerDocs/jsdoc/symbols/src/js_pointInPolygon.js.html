  8 /*
  9  * Detect if a given point is inside a polygon
 10  *
 11  * @param point the point in [x,y] format
 12  * @param vs vertices in [[x1,y2],[x2,y2],...] format
 13  */
 14 function pointInPolygon(point, vs) {
 15     var x = point[0], y = point[1];
 16     
 17     var inside = false;
 18     for (var i = 0, j = vs.length - 1; i < vs.length; j = i++) {
 19         var xi = vs[i][0], yi = vs[i][1];
 20         var xj = vs[j][0], yj = vs[j][1];
 21         
 22         var intersect = ((yi > y) != (yj > y))
 23             && (x < (xj - xi) * (y - yi) / (yj - yi) + xi);
 24         if (intersect) inside = !inside;
 25     }
 26     
 27     return inside;
 28 }
