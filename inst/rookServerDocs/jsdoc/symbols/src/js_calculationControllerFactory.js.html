  2 /**
  3  * Factory class for calculation controller
  4  * @description the calculation controller does not have to be a singleton
  5  * but all the objects need to be intialised with the repository
  6  * parameters specified in the initialise() function. This factory is 
  7  * a singleton that will do this while allowing multiple calculationController 
  8  * objects to coexist
  9  * @constructor
 10  * @repos {string} repository type
 11  */ 
 12 function calculationControllerFactory(repos) {
 13     if ( typeof calculationControllerFactory.instance === 'object' ){
 14 	return calculationControllerFactory.instance;
 15     }
 16 
 17     this.repository  = repos;
 18 
 19     calculationControllerFactory.instance = this;
 20 }
 21 
 22 calculationControllerFactory.prototype.makeCalculationController = function() {
 23    return new calculationController(this.repository, true);
 24 }

 33  * @constructor

 35 function calculationController(computeEngineType, initialisedByFactory) {
 36     if (initialisedByFactory !== true) {
 37 	throw('Calculation controller cannot be initialised directly! Use calculationControllerFactory.');

 39 
 40     this.computeEngine = null;
 41 
 42     if (computeEngineType === 'local') {
 43 	this.computeEngine =  new localComputeEngine();
 44     } else if (computeEngineType === 'remote') {
 45 	this.computeEngine = new remoteComputeEngine();
 46     }  else {
 47 	console.error("Unknown compute engine type: " + computeEngineType);
 48     }
 49 }

115 /**
116  * Winsorize Matrix row by row
117  * @description winsorize matrix row by row clipping 'trim' extreme values
118  * this implementation is rather slow, but it does generate deep copy of the matrix
119  * @parameter matrix a 2d matrix with all rows of equal lenght
120  * @parameter trim percent of extreme values to clip from both ends
121  * @parameter skipColumns number of columns to skip on the right
122  * @returns the winsorized matrix
123  * @obsolete 
124  */
125 localComputeEngine.prototype.winsorizeMatrix = function(matrix, trim, skipColumns) {
126     var retMatrix = [];
127 
128     var k = matrix.length; // n rows
129     var n = matrix[0].length; // n cols
130 
131     // Number to trim
132     var ntr = Math.round(n * trim);
133     
134     // Row
135     for ( var i = 0 ; i < k; i++) {
136 	var row = matrix[i];
137 	var newRow = [];
138 
139 	var rowOrdered = row.slice();
140 	rowOrdered.sort();
141 	
142 	// Set limits
143 	var minv = rowOrdered[ntr] ;
144 	var maxv = rowOrdered[n - ntr -1];
145 
146 	for (var j = 0; j < n; j++) {
147 	    if (j < skipColumns) {
148 		// skipColumns -- these are usually labels
149 		newRow[j] = row[j];;
150 	    } else {
151 		if (row[j] < minv) {
152 		    newRow[j] =  minv;
153 		} else if (row[j] > maxv) {
154 		    newRow[j] = maxv;
155 		} else {
156 		    newRow[j] = row[j]
157 		}
158 	    }
159 	} // for var j
160 
161 	retMatrix[i] = newRow;
162     }
163 
164     return retMatrix;
165 }
