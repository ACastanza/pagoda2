  7 /**
  8  * implements the main event bus
  9  * @constructor
 10  * @description it is used to communicate between
 11  * different components on the page without having to create
 12  * binding code. Implements Singlenton.
 13  * @example
 14      // Example usage
 15      var evtBus = new eventBus();
 16      
 17      // Register an event handler
 18      evtBus.register("customEventIdentifier", 
 19                      {info: "Some information From Registration Time"},
 20 		     function(event, params) {
 21 		           // Registation information in e.data
 22 			   // Publish information in params
 23 		     });
 24      // Raise an event
 25      evtBus.publish("customEventIdentifier",
 26                     {info: "Some information from publicationtime"});
 27  */
 28 function eventBus() {
 29     if (typeof eventBus.instance === 'object') {
 30 	return eventBus.instance;
 31     }
 32 
 33     eventBus.instance = this;
 34 };
 35 
 36 /**
 37  * register an event
 38  */
 39 eventBus.prototype.register = function(event, data, handler) {
 40 	$(this).on(event, null, data, handler);
 41 };
 42 
 43 /**
 44  * unregister an event
 45  */
 46 eventBus.prototype.unregister = function(event, data, handler) {
 47 	$(this).off(event, data, handler);
 48 };
 49 
 50 /**
 51  * publish an event
 52  */
 53 eventBus.prototype.publish = function(event, params) {
 54 	$(this).trigger(event, params);
 55 };
