 35     this.geneRegions = new clickableRegions();

 63     heatmapAreaOverlay.css({
 64 	position: 'absolute',
 65 	top: 0,
 66 	left: 0,
 67     });
 68

 71

 73     this.updateCanvasSize();

 76     this.displayGenes = [];

 84     this.drawHeatmap();

159 /**
160  * Get the order in which the rows of the provided data
161  * are best ordered in for visual purposes
162  * @param data a dgCMatrix reader with the sparse array data
163  */
164 heatmapViewer.prototype.getRowVisualOrder = function(data) {
165 //    the bin size
166 
167 
168     var ncols = data.Dim[0];
169     var nrows = data.Dim[1];
170 
171 
172     var binsize = Math.max(ncols / 30);
173     var ncolsbin = Math.ceil(ncols / binsize);
174 
175     // Make empty bins array
176     var binsSumArray = new Array(nrows);
177     for (var k = 0; k < nrows; k++) {
178     	var row = new Array(ncolsbin);
179     	for (var j = 0; j < ncolsbin; j++) {
180     	    row[j] = 0;
181     	}
182     	binsSumArray[k] = row;
183     }
184 
185     // Sum values for every bin
186     for (var j = 0; j < data.p.length - 1; j++) {
187     	// Row start and end index
188     	var rsi = data.p[j];
189     	var rei = data.p[j+1] - 1;
190 
191     	for (var k = rsi; k < rei; k++) {
192     	    var cn = data.i[k];
193 
194     	    // Find the bin index for this element
195     	    var binindex = Math.floor(cn / binsize);
196     	    binsSumArray[j][binindex] = binsSumArray[j][binindex] + data.x[k];
197     	}
198     }
199 
200     function meanClampNorm(v, rowMean, maxAbsValue) {
201     	const trim = 0.3;
202     	maxAbsValue *= trim;
203 
204     	const range = 1;
205 
206     	var val =  (v - rowMean) / (maxAbsValue * 2) + 0.5;
207     	val = val * range;
208     	val = val < 0 ? 0 : val;
209     	val = val > range ? range : val;
210     	return val;
211     }
212 
213     // Normalise
214     for (var j = 0; j < binsSumArray.length; j++) {
215 
216     	var maxAbsValue = 0;
217 
218     	var rowMean = 0;
219     	var row = binsSumArray[j];
220     	for (var i = 0; i < row.length; i++) {
221     	    rowMean += row[i];
222     	    maxAbsValue =  Math.abs(row[i]) > maxAbsValue ? Math.abs(row[i]) : maxAbsValue;
223     	}
224     	rowMean /= row.length;
225 
226     	for (var i =0; i< row.length; i++) {
227     	    row[i] = meanClampNorm(row[i], rowMean, maxAbsValue);
228     	}
229 
230     	// This is probably not requried
231     }
232 
233     d = binsSumArray;
234 
235     for (var j = 0; j < d.length; j++) {
236 	var maxAbsValue = 0;
237 	var rowMean = 0;
238 	for (var i = 0; i < d[j].length; i++) {
239 	    rowMean += d[j][i];
240 	    maxAbsValue = Math.abs(d[j][i]) > maxAbsValue ? Math.abs(d[j][i]) : maxAbsValue;
241 	}
242 	rowMean /= d[j].length;
243 	for (var  i =0; i < d[j].length; i++) {
244 	    d[j][i] = meanClampNorm(d[j][i], rowMean, maxAbsValue);
245 	}
246     }
247 
248     // Do hierarchical clustering
249     hc = hcluster(d, pagHelpers.seq(0, d.length - 1), 'corrdist', 'average')
250     // Do a depth first search on the tree
251     function getOrder(hc) {
252 	var order = [];
253 	if (hc.hasOwnProperty('left')) {
254 	    order = order.concat(getOrder(hc.left));
255 	    order = order.concat(getOrder(hc.right));
256 	} else {
257 	    order.push(hc.label);
258 	}
259 	return order;
260     }
261     order = getOrder(hc);
262 
263     // Change into the required format
264     var order2 = [];
265     for (var i = 0; i < order.length; i++) {
266 	order2[order[i]] = i;
267     }
268 
269 
270     // Return the order from the depth first
271     return order2;
272 
273 }

288 /**
289  * Show the overlay crosshairs and tooltip
290  */
291 heatmapViewer.prototype.showOverlay = function (x,y, label) {
292 
293     // Avoid jquery lookup for performance here
294     var overlayArea = document.getElementById('heatmap-area-overlay');
295     var ctx = overlayArea.getContext('2d');
296 
297     var heatView = new heatmapViewer();
298 
299     var drawConsts = heatView.getDrawConstants();
300 
301     var areaWidth = overlayArea.width;
302     var areaHeight = overlayArea.height;
303 
304     ctx.setLineDash([10,10])
305     ctx.lineWidth = 1;
306     ctx.clearRect(0,0,areaWidth,areaHeight);
307 
308     var actualPlotHeight = heatView.getActualPlotHeight();
309 
310     if (typeof y !== 'undefined' & y < actualPlotHeight
311        ){
312 	ctx.beginPath();
313 	ctx.moveTo(drawConsts.left, y);
314 	ctx.lineTo(drawConsts.width + drawConsts.left, y);
315 	ctx.stroke();
316     }
317 
318     if (typeof x !== 'undefined' &
319 	x > drawConsts.left &
320 	x < drawConsts.width + drawConsts.left  &
321 	(y < actualPlotHeight  | typeof y === 'undefined') // if y is provided it is in the plot
322        ) {
323 
324 	ctx.beginPath();
325 	ctx.moveTo(x, drawConsts.top);
326 	ctx.lineTo(x, actualPlotHeight + drawConsts.top);
327 	ctx.stroke();
328     }
329 
330     // The tooltip
331     if (typeof y !== 'undefined' & typeof x !== 'undefined' & typeof label !== 'undefined') {
332 
333 	ctx.save();
334 	const tooltipTop = 30;
335 	const tooltipLeft = 30;
336 
337 	const tooltipWidthPadding = 20;
338 	const tooltipHeight = 30;
339 
340 	const tooltipShadowOffset = 3;
341 
342 	ctx.font = '18px Arial';
343 	var textSize = ctx.measureText(label);
344 
345 	// Tooltip text
346 	ctx.fillStyle = 'black';
347 	ctx.fillText(label, x + tooltipLeft , y + tooltipTop);
348 	ctx.restore();
349     }
350 }

424 /**
425  * Draws the heatmap
426  */
427 heatmapViewer.prototype.drawHeatmap = function() {
428     if (typeof this.displayGenes === 'undefined' || this.displayGenes.length === 0) {
429 	// No gene selected
430 
431 	ctx = this.getDrawingContext();
432 	this.clearHeatmap(ctx);
433 
434 	var heatDendView = new heatmapDendrogramViewer();
435 
436 	var left = heatDendView.getPlotAreaLeftPadding();
437 	var heatmapWidth = heatDendView.getPlotAreaWidth();
438 	var heatmapHeight = heatDendView.getCurrentHeatmapHeight();
439 
440 	const text = 'No Genes selected';
441 
442 	ctx.font = '20px Arial';
443 
444 	// Center justify the lest
445 	var measure = ctx.measureText(text);
446 	left = left - measure.width /2;
447 	ctx.fillText(text, heatmapWidth /2 + left, heatmapHeight/2);
448 
449 
450 	// We also want to maintain a gene selection showing the
451 	// current plotted genes
452 	var geneSelCntr = new geneSelectionController();
453 	geneSelCntr.setSelection('heatmapDisplayGenes', [], 'Current Heatmap Genes');
454     } else {
455 
456 	// Make a gene selection with the currently shown genes
457 	var geneSelCntr = new geneSelectionController();
458 	geneSelCntr.setSelection('heatmapDisplayGenes', this.displayGenes, 'Current Heatmap Genes');
459 	this.doDrawHeatmap();
460     }
461 }

514 /**
515  * Internal function for drawing the heatmap using the sparse matrix directly
516  * @description assumes that the displayGenes in is not empty
517  * @private
518  */
519 heatmapViewer.prototype.doDrawHeatmapSparseMatrix = function() {
520 
521     var dendV = new dendrogramViewer();
522     var heatDendView = new heatmapDendrogramViewer();
523 
524 
525 // heatDendView.getCurrentHeatmapHeight()
526 
527     // Get the drawing params
528     drawConsts = this.getDrawConstants();
529     var top = drawConsts.top;
530     var left = drawConsts.left;
531     var heatmapWidth = drawConsts.width;
532     var heatmapHeight = drawConsts.height - drawConsts.paddingBottom;
533 
534     // Get the genes to plot
535     var geneSelection = this.displayGenes;
536 
537     // Get the cells to plot
538     var cellRange = dendV.getCurrentDisplayCellsIndexes();
539     var cellIndexStart = cellRange[0];
540     var cellIndexEnd = cellRange[1];
541 
542     var heatView = this;
543 
544     // Clear heatmap
545     var ctx = heatView.getDrawingContext();
546     heatView.clearHeatmap(ctx);
547 
548     // Show centered waiting icon
549     $('#heatmap-area-container').append("<img class='loadingIcon' src='img/loading.gif'/>");
550     var loadingDomItem =  $('#heatmap-area-container > .loadingIcon')
551     var lpad = heatDendView.getCurrentWidth()  / 2;
552     var tpad = heatDendView.getCurrentHeatmapHeight() /2;
553     loadingDomItem.css({'padding-left': lpad + 'px', 'padding-top': tpad + 'px'});
554 
555     var dataCntr = new dataController();
556     dataCntr.getExpressionValuesSparseByCellIndexUnpacked(geneIds = geneSelection, cellIndexStart = cellIndexStart, cellIndexEnd = cellIndexEnd, getCellNames = false, callback =  function(data) {
557         loadingDomItem.remove();
558 
559 	// Get the dimensions
560 	var ngenes = data.Dim[1];
561 	var ncells = data.Dim[0];
562 
563 	// Computed plotting params
564 	var cellWidth = heatmapWidth / ncells;
565 	var cellHeight = heatmapHeight / ngenes;
566 
567 	// Limit max cell height
568 	drawConsts = heatView.getDrawConstants(); // Do we need this again
569 	cellHeight = Math.min(cellHeight, drawConsts.maxCellHeight);
570 
571 	// Calculate Height after taking max cell size to account
572 	var actualPlotHeight = ngenes * cellHeight;
573 
574 	// Have to set it on the obj for the crosshairs
575 	heatView.setActualPlotHeight( actualPlotHeight );
576 
577 	// Get a new row order
578 	var rowOrder;
579 	if (heatView.getRowReordering() === true) {
580             rowOrder = heatView.getRowVisualOrder(data);
581 	} else {
582 	    rowOrder = pagHelpers.seq(0,data.Dim[1] -1);
583 	}
584 
585 	// Get palette
586 	var palSize = heatView.palManager.getNumberOfColors();
587 	var pal = heatView.palManager.getPaletteColors();
588 
589 	// Plot background
590 	ctx.fillStyle = pal[Math.floor(palSize/2)];
591 	ctx.fillRect(left,top,heatmapWidth,actualPlotHeight);
592 
593 	for ( var j = 0; j < data.p.length - 1; j++) {
594 	    // row start index, row end index (in x and i)
595 	    var rsi = data.p[j];
596 	    var rei = data.p[j+1] -1;
597 
598 	    // Calculate row normalisation values
599 	    var rowMin = data.x.slice(rsi, rei).reduce(function(a,b){ return Math.min(a,b) } );
600 	    var rowMax = data.x.slice(rsi, rei).reduce(function(a,b){ return Math.max(a,b) } );
601 	    var rowSum = data.x.slice(rsi, rei).reduce(function(a,b){ return a+b });
602 	    var rowMean = rowSum / (rei - rsi + 1);
603 	    var maxAbsValue = Math.max(Math.abs(rowMin - rowMean), Math.abs(rowMax - rowMean));
604 
605 	    // color mapper is a function
606 	    // use a color mapper to ensure consistency of coloring with
607 	    // other views (eg embedding)
608 	    var colorMapper = heatView.palManager.getMeanClampedColorMapper(rowMean, maxAbsValue, palSize);
609 
610 	    // Plot row
611 	    for (var k = rsi; k < rei; k++) {
612 //		var plotValue = (data.x[k] - rowMean) / (maxAbsValue * 2) + 0.5;
613 //		var palIndex = Math.floor(plotValue * (palSize)) - 1;
614 
615 		var palIndex = colorMapper(data.x[k]);
616 		ctx.fillStyle = pal[palIndex];
617 
618 		var x = data.i[k] * cellWidth + left;
619 		var y = rowOrder[j] * cellHeight + top; // reorder on the fly
620 
621 		ctx.fillRect(x,y,
622 		    cellWidth, cellHeight);
623 	    } // for k
624 
625 	} // for j
626 
627 	// Draw bounding box
628 	ctx.strokeRect(left, top, heatmapWidth, actualPlotHeight);
629 
630 	// Plot the labels
631 	for (var i = 0; i < data.DimNames2.length; i++) {
632 	    // Print the names
633 	    var name = data.DimNames2[i];
634 
635 	    // Get the appropriate font size for this number of cells
636 	    var fontSize =  heatView.getRowFontSize(cellHeight)
637 
638 	    // Calculate position
639 	    x = ncells * cellWidth + left + 20;
640 	    y = rowOrder[i] * cellHeight + top + cellHeight / 2 + fontSize / 3;
641 
642 	    // Plot
643 	    ctx.font =  fontSize + 'px Arial';
644 	    ctx.fillStyle = 'black';
645 	    ctx.fillText(name, x, y);
646 	} // Label plot loop
647 
648 
649 	// Setup the resolvable regions (to gene names)
650 	// these are used for tooltips and coloring by gene
651 	heatView.geneRegions.clearClickAreas();
652 	for (var i = 0; i < data.DimNames2.length; i++) {
653 	    j = rowOrder[i];
654 
655 	    var x1 = left;
656 	    var y1 = cellHeight * j + top;
657 	    var x2 = heatmapWidth;
658 	    var y2 = cellHeight * ( j +1 ) + top;
659 
660 	    heatView.geneRegions.addClickArea(
661 		x1, y1,
662 		x1, y2,
663 		x2, y2,
664 		x2, y1,
665 		{geneId: data.DimNames2[i] }
666 	    );
667 	} // for
668 
669 
670     }); // dataCntr.getExpressionValuesSparseTransposedByCellIndexUnpacked callback
671 } // doDrawHeatmapSparseMatrix
