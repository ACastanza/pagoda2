  2 /*
  3  *
  4  * The selection cotroller is responsible for keeping 
  5  * track of cell selections. The selections are kept as 
  6  * names objects in an array and they are themselves arrays of 
  7  * text labels of cells
  8  *
  9  * Some selection names have special meanings, but this is
 10  * not otherwise enforced:
 11  * 
 12  * currentPrimarySel -- the current selection of cells
 13  * currentSecondarySel -- the current secondary selection of cells 
 14  *                        when one is needed, such as when selecting
 15  *                        two dendrogram branches
 16  *
 17  */
 18 
 19 /**
 20  * Keeps track of cell seletions
 21  * @constructor
 22  */
 23 function cellSelectionController() {
 24     if (typeof cellSelectionController.instance === 'object') {
 25 	return cellSelectionController.instance;
 26     }
 27 
 28     // Generat the array to keep the selections in 
 29     this.selections = new Object();
 30 
 31     cellSelectionController.instance = this;
 32 };
 33 
 34 /**
 35  * Sets a cell selection
 36  * @param {string} selectionName A name for this cell selection
 37  * @param {Array[]} cells identifiers of the cells for this selection
 38  * @param {string} displayName Name to show in the UI when refering to this selection
 39  * @param {object} metadata any kind of metadata we want to attach to this cell selection
 40  */
 41 cellSelectionController.prototype.setSelection = function(selectionName, cells, displayName, metadata ) {
 42      if (typeof displayName === 'undefined') {
 43 	displayName = selectionName;
 44     }
 45     
 46     this.selections[selectionName] = {
 47 	'name': selectionName,
 48 	'cells': cells,
 49 	'displayName': displayName,
 50 	'metadata': metadata
 51     };
 52 
 53     this.raiseSelectionChangedEvent();
 54 };
 55

 57 /**
 58  * Raise an event for notifying that the cell selectiion has changed
 59  * @private
 60  */
 61 cellSelectionController.prototype.raiseSelectionChangedEvent = function() {
 62     // TODO: buffer this
 63       // Notify the world that a new selection has been set
 64     var evtBus = new eventBus();
 65     evtBus.publish("cell-selection-updated");  
 66 }

 68 /**
 69  * Get the names of the currently available selections
 70  */
 71 cellSelectionController.prototype.getAvailableSelections = function() {
 72     var names = [];
 73     for (sel in this.selections) {
 74 	names.push(this.selections[sel].name);
 75     }
 76 
 77     return names;
 78 };
 79 
 80 /**
 81  * Delete a cell selection
 82  * @param {string} selectionName the name of the selection to delete
 83  */
 84 cellSelectionController.prototype.deleteSelection = function(selectionName) {
 85     delete this.selections[selectionName];
 86 }
 87 
 88 /**
 89  * Get a cell selection
 90  * @param {string} selectionName the internal name of the cell selection
 91  * @returns {Array[]} Identifiers of the cells in this selection 
 92  */
 93 cellSelectionController.prototype.getSelection = function(selectionName) {
 94     var sel =  this.selections[selectionName];
 95     if (typeof sel !== 'undefined') {
 96 	return sel.cells;
 97     } else {
 98 	return undefined;
 99     }
100 };
101 
102 /**
103  * get the display name of a cell selection
104  * @param selectionName the internal name of the selection
105  * @retuns the display name of the selection
106  */
107 cellSelectionController.prototype.getSelectionDisplayName = function(selectionName) {
108     var sel =  this.selections[selectionName];
109     if (typeof sel !== 'undefined') {
110 	return sel.displayName;
111     } else {
112 	return undefined;
113     };
114 };
115 
116 /**
117  * Create a new cell selection by duplicating an existing one
118  * @param selectionName {string} name of the selection to duplicate
119  * @param newSelectionName {string} name of the new selection to make
120  * @param newSelectionDisplayName {string} display name of the new selection
121  */
122 cellSelectionController.prototype.duplicateSelection = function(selectionName,
123 								newSelectionName,
124 								newSelectionDisplayName) {
125     var oldSelection = this.selections[selectionName];
126     var sel = {};
127     sel.name = newSelectionName;
128     sel.displayName = newSelectionDisplayName;
129 
130     sel.cells = JSON.parse(JSON.stringify(oldSelection.cells));
131     
132     this.selections[newSelectionName] = sel;
133     this.raiseSelectionChangedEvent();
134 }

136 /**
137  * Rename a cell selection, only change the display name
138  */
139 cellSelectionController.prototype.renameSelection = function(selectionName, newSelectionName) {
140     // TODO check tha the new selection name does not already exist
141     this.selections[selectionName].displayName =  newSelectionName;
142     this.raiseSelectionChangedEvent();
143 }
