397 /**
398  * Get the parent node of the specified internal node
399  * @nodeId the internal node id, 1 indexed
400  */
401 dendrogramViewer.prototype.getInternalNodeParent = function(nodeId, callback) {
402     var dataCntr = new dataController();
403     dataCntr.getReducedDendrogram(function(data) {
404 	var retValue = undefined;
405 
406 	var n = data.order.length;
407 	mergeArray = pagHelpers.serialisedArrayTo2D(data["merge"], n - 1, 2);
408 
409 	if (nodeId >= n - 1) {
410 	    // We are at the head node
411 	} else {
412 	    for (var i = 0; i < n; i++) {
413 		if (mergeArray[i][0] == nodeId || mergeArray[i][1] == nodeId) {
414 		    retValue = i + 1;
415 		    break; // Stop searching
416 		};
417 	    };
418 	    callback(retValue);
419 	}
420     });
421 }

423 /**
424  * Get an object with all the leafs under this node
425  * @param data The hclust data
426  * @param nodeId The nodeId the terminal leafs of which to get, 1-indexed as in the merge matrix
427  */
428 dendrogramViewer.prototype.getNodeDescendents =  function(data, nodeId) {
429     var n = data.order.length;
430     var mergeArray = pagHelpers.serialisedArrayTo2D(data["merge"], n - 1, 2);
431 
432     var pendingVisit = [];
433     var children = [];
434 
435     // Starting point
436     pendingVisit.push(nodeId);
437 
438     while(pendingVisit.length > 0) {
439 	var currentNode = pendingVisit.pop();
440 	if (currentNode < 0) {
441 	    // It's a leaf
442 	    // Add 1 so that we don't change the element s
443 	    // Was was subtracted when they were pushed
444 	    children.push(Math.abs(currentNode));
445 	} else {
446 	    // It's an internal node visit its children
447 	    // Subtract 1 because the numbers in the merge array
448 	    // Refer to 1-indexes R positions
449 	    pendingVisit.push(mergeArray[currentNode - 1][0]);
450 	    pendingVisit.push(mergeArray[currentNode - 1][1]);
451 	}
452     }
453     return children;
454 }

457 /**
458  * Updates the transform matrix using the built in
459  * transform function and keeps track of the
460  * applied transformations
461  */
462 dendrogramViewer.prototype.transform = function(a,b,c,d,e,f) {
463     //dendArea = $('#dendrogram-area')[0];
464     //var ctx = dendArea.getContext("2d");
465     // Read here for details
466     // https://developer.mozilla.org/en-US/docs/Web/API/Canvas_API/Tutorial/Transformations
467 
468     // Update the locally kept transform matrix
469     var dendV = new dendrogramViewer();
470     var newTransformMatrix = [ [a, c, e], [b, d, f], [0, 0, 1] ];
471     // Multiplication is done the other way around to
472     // keep in line with the orientation of the matrices
473     // that canvas maintains
474     dendV.currentTransform = (math.multiply(newTransformMatrix, dendV.currentTransform));
475     dendV.restoreCurrentTransform();
476 };

501 /**
502  * Transforms an x and y from the canvas back to the plot coordinates
503  * @description Takes an x and y from the canvas and
504  * takes it back through the applied transformation to
505  * find which position in the canvas plotting coordinate
506  * frame was clicked
507  * @param x the x-coordinate
508  * @param u the y-coordinate
509  */
510 dendrogramViewer.prototype.transformToLocalCoordinates = function(x,y) {
511     var dendV = new dendrogramViewer();
512 
513     // Augment the point
514     var p = [x, y, 1];
515     var invTr = dendV.getCurrentInverseTransform();
516 
517     // Map back through current transform
518     var np = math.multiply(invTr, math.transpose(p) );
519 
520     // Return x and y only
521     return np.slice(0,2);
522 };
523 
524 /**
525  * Transform to canvas plot to regular canvas coordinates
526  */
527 dendrogramViewer.prototype.transformToCanvasCoordinates = function(x,y) {
528     var p = [x,y,1];
529     var tr = this.getCurrentTransform();
530     var np = math.multiply(tr, math.transpose(p));
531     return np.slice(0,2);
532 }

534 /**
535  * Put the current transform that we have in our matrix
536  * into the canvas object
537  */
538 dendrogramViewer.prototype.restoreCurrentTransform = function() {
539      var dendV = new dendrogramViewer();
540     var curT = dendV.currentTransform;
541     a = curT[0][0];
542     b = curT[2][0];
543     c = curT[0][1];
544     d = curT[1][1];
545     e = curT[0][2];
546     f = curT[1][2];
547 
548     // Transform main canvas
549     dendArea = $('#dendrogram-area')[0];
550     var ctx = dendArea.getContext("2d");
551     ctx.setTransform(a,b,c,d,e,f);
552 
553     // Transform the overlay canvas as well
554     dendAreaOverlay = $('#dendrogram-area-overlay')[0];
555     var ctx2 = dendAreaOverlay.getContext("2d");
556     ctx2.setTransform(a,b,c,d,e,f);
557 
558     // Transform the internal copy
559     var ctx3 = this.canvasCopy.getContext('2d');
560     ctx3.setTransform(a,b,c,d,e,f);
561 
562     var ctx4 = this.canvasCopy2.getContext('2d');
563     ctx4.setTransform(a,b,c,d,e,f);
564 }

612 /**
613  * Given the order of the clusters and the number of items
614  * they contain return an array with the start and end positions
615  * of the clusters as they are ploted.
616  * @param data the data from the server side
617  */
618 dendrogramViewer.prototype.getClusterPositionRanges =  function (data) {
619     // TODO: this can in principle be cached
620 
621     var memCountInOrder = [];
622 
623     // Put them in order (start pos depends on what comes before)
624     for (var i = 0; i < data.order.length; i++) {
625 	memCountInOrder[i] = data.clusterMemberCount[data.order[i]];
626     }
627 
628     // Calculate the coordinates
629     var coordInOrder = [];
630     var lastPos =0;
631     for (var i =0; i < memCountInOrder.length; i++) {
632 	var start = lastPos + 1;
633 	var end = start + memCountInOrder[i] - 1;
634 	coordInOrder[i] = [start, end];
635 	lastPos = end;
636     }
637 
638     // Put the coordinates back in order
639     var coordOfNodes = [];
640     for (var i = 1; i<= data.order.length; i++) {
641 	var indx = data.order.indexOf(i);
642 	coordOfNodes[i] = coordInOrder[indx];
643     }
644 
645     return coordOfNodes;
646 }

674 /**
675  * Draw a dendrogram that only includes everything under the specified node
676  * @param data the data as received from the dataController
677  * @param topnode the identifier for the top most node to draw
678  */
679 dendrogramViewer.prototype.drawDendrogramSubsetWithData = function (data, topnode) {
680     // Get the dendrogram viewer object
681     var dendV =  new dendrogramViewer();
682     var heatDendView = new heatmapDendrogramViewer();
683 
684     // These are the leaf node ids that are below our topnode
685     var children = this.getNodeDescendents(data, topnode);
686 
687     // Recover the data
688     var n = data.order.length;
689     mergeArray = pagHelpers.serialisedArrayTo2D(data["merge"], n - 1, 2);
690 
691     // Max height is  the height of the top node
692     var maxHeight = data.height[topnode - 1];
693 
694     // Vertical scaling
695     var vscale = heatDendView.getCurrentDendHeight() / maxHeight *
696 	p2globalParams.dendrogram.scaleFactor;
697 
698     // Get the canvas object
699     var dendAreaOuter = $('#dendrogram-area');
700     dendArea = dendAreaOuter[0];
701     var ctx = dendArea.getContext("2d");
702     ctx.save();
703 
704     // Set line width
705     ctx.lineWidth = this.currentConfiguration.lineWidth;
706 
707     // Clear the canvas
708     ctx.clearRect(0,0,heatDendView.getCurrentWidth(),heatDendView.getCurrentDendHeight())
709 
710     // Calculating leaf positions
711     coordOfNodes = dendV.getClusterPositionRanges(data);
712 
713 
714 
715     // Get the mid-points of the  clusters
716     var nodePositions = [];
717     for (var i = 1; i < coordOfNodes.length; i++) {
718 	nodePositions[i] = [(coordOfNodes[i][0] + coordOfNodes[i][1]) /2, 0];
719     }
720 
721     // Get Min/max only for the children of the nodes we are plotting
722     scaleMin = +Infinity;
723     scaleMax = -Infinity;
724     for ( var i = 1; i < coordOfNodes.length; i++) {
725 	if (  children.indexOf(i) >= 0 ) {
726 	    if ( coordOfNodes[i][0]  < scaleMin ) {
727 		scaleMin = coordOfNodes[i][0];
728 	    }
729 	    if ( coordOfNodes[i][1] > scaleMax ) {
730 		scaleMax = coordOfNodes[i][1];
731 	    }
732 	}
733     }
734 
735     // Update the configuration of the dendrogram with
736     dendV.currentConfiguration.startCellIndex = scaleMin - 1;
737     dendV.currentConfiguration.endCellIndex = scaleMax - 1;
738 
739     // This updates the current cell order (by identifier) and raises the event.
740     dendV.updateCurrentDisplayCells(function() {
741 	var evtBus = new eventBus();
742 	evtBus.publish('dendrogram-cell-order-updated');
743     });
744 
745     // Calculate the scallings for the transformation
746     var horizontalScaling = heatDendView.getPlotAreaWidth() / (scaleMax - scaleMin + 1);
747     var horizontalShift = -1 * scaleMin * horizontalScaling +
748 	heatDendView.getPlotAreaLeftPadding();
749 
750 
751     // Two helper function for doing the transformation before plotting
752     // instead of using the canvas.
753     function scaleX(x) { return x * horizontalScaling + horizontalShift; }
754     function scaleY(y) { return y * vscale; }
755 
756     // Set cartesian coordinates on the canvas
757     dendV.setTransform(1,0,0,-1,0,dendArea.height);
758 
759     // Clear the current Click Areas
760     dendV.clickRegions.clearClickAreas();
761 
762     // An array that holds the internal positions in [x,y] format
763     // This is needed so that when a merge refers to a previous
764     // merge the plot start can be easily determined
765     var internalNodePositions = [];
766 
767     // This array holds [x1, x2] range of x-coordinates that
768     // are to be highlighted for each internal node
769     var internalNodeOverlayRanges = [];
770 
771     // Loop over merge array
772     for (var i = 0; i <  mergeArray.length; i++) {
773 	// Keeping track of node boundaries
774 	var currentNodeLeft = +Infinity;
775 	var currentNodeRight = -Infinity;
776 
777 	// Leaf or internal
778 	var nodeType = null;
779 
780 	// Height for this step
781 	var drawHeight = data.height[i];
782 
783 	// Merge array entry
784 	var mergeStep = mergeArray[i];
785 
786 	// The two points to connect with a horisontal line in the last step
787 	var p1 = undefined;
788 	var p2 = undefined;
789 
790 	// Step 1:
791 	// draw line for node1 to appropriate height
792 	// register this line with the click array
793 	if (mergeStep[0] < 0) {
794 	    // A leaf node (-ve), check if it is in the children array
795 	    var nodeId = Math.abs(mergeStep[0]);
796 	    nodeType = "leaf";
797  	    if (children.indexOf(nodeId) >= 0) {
798 		// Keep track of left and right for this node
799 		currentNodeLeft = Math.min(currentNodeLeft, Math.min(coordOfNodes[nodeId][0],
800 								     coordOfNodes[nodeId][1]) );
801 		currentNodeRight = Math.max(currentNodeRight, Math.max(coordOfNodes[nodeId][0],
802 								       coordOfNodes[nodeId][1]) );
803 
804 		p1 = nodePositions[nodeId];
805 
806 		ctx.beginPath();
807 		ctx.moveTo(scaleX(p1[0]),scaleY(0));
808 		ctx.lineTo(scaleX(p1[0]),scaleY(drawHeight));
809 		ctx.stroke();
810 
811 		// No click region for nodes
812 	    }
813 	} else {
814 	    // An internal node
815 	    var nodeId = mergeStep[0];
816 	    nodeType = "internal";
817 	    var intNodePos = internalNodePositions[nodeId];
818 
819 	    // It is possible that the internal node does not
820 	    // exist because its descendents are not part of
821 	    // what we are currently plotting
822 	    if (typeof intNodePos !== 'undefined') {
823 
824 		currentNodeLeft = Math.min(currentNodeLeft, Math.min(internalNodeOverlayRanges[nodeId][0],
825 					   internalNodeOverlayRanges[nodeId][1]) );
826 		currentNodeRight = Math.max(currentNodeRight, Math.max(internalNodeOverlayRanges[nodeId][0],
827 								       internalNodeOverlayRanges[nodeId][1]));
828 
829 		// Draw vertical line
830 		ctx.beginPath();
831 		ctx.moveTo(scaleX(intNodePos[0]),scaleY(intNodePos[1]));
832 		ctx.lineTo(scaleX(intNodePos[0]),scaleY(drawHeight));
833 		ctx.stroke();
834 
835 		var x1 = scaleX(intNodePos[0]) - 2;
836 		var y1 = scaleY(intNodePos[1]) - 2;
837 		var x2 = scaleX(intNodePos[0]) + 2;
838 		var y2 = scaleY(drawHeight) +2;
839 
840 		var left = scaleX(Math.min(internalNodeOverlayRanges[nodeId][0],internalNodeOverlayRanges[nodeId][1]));
841 		var width = scaleX(Math.max(internalNodeOverlayRanges[nodeId][0],internalNodeOverlayRanges[nodeId][1])) - left;
842 
843 
844 		dendV.clickRegions.addClickArea(
845 		    x1, y1,
846 		    x1, y2,
847 		    x2, y2,
848 		    x2, y1,
849 		    {
850 			nodeId: nodeId,
851 			nodeType: 'vertical',
852 			highlightBoundingbox: {
853 			    top: 0,
854 			    left: left,
855 			    width: width,
856 			    height: scaleY(drawHeight) - this.currentConfiguration.lineWidth * 2
857 			}
858 		    }
859 		);
860 
861 
862 		p1 = intNodePos;
863 	    }
864 	}
865 
866 	// Step 2:
867 	// draw line for node2 to appropriate height
868 	// register this line with the click array
869 	if (mergeStep[1] < 0) {
870 	    var nodeId = Math.abs(mergeStep[1]);
871 	    nodeType = "leaf";
872 	    if (children.indexOf(nodeId) >= 0) {
873 		p2 = nodePositions[nodeId];
874 
875 		// Keep track of left and right for this node
876 		currentNodeLeft = Math.min(currentNodeLeft, Math.min(coordOfNodes[nodeId][0],
877 								     coordOfNodes[nodeId][1]) );
878 		currentNodeRight = Math.max(currentNodeRight, Math.max(coordOfNodes[nodeId][0],
879 								       coordOfNodes[nodeId][1]) );
880 
881 		ctx.beginPath();
882 		ctx.moveTo(scaleX(p2[0]),scaleY(0));
883 		ctx.lineTo(scaleX(p2[0]),scaleY(drawHeight));
884 		ctx.stroke();
885 	    }
886 	} else {
887 	    // An internal node
888 	    var nodeId = mergeStep[1];
889 	    nodeType = "internal";
890 	    var intNodePos = internalNodePositions[nodeId];
891 
892 	    if (typeof intNodePos !== 'undefined') {
893 		currentNodeLeft = Math.min(currentNodeLeft, Math.min(internalNodeOverlayRanges[nodeId][0],
894 								      internalNodeOverlayRanges[nodeId][1]));
895 		currentNodeRight = Math.max(currentNodeRight, Math.max(internalNodeOverlayRanges[nodeId][0],
896 								       internalNodeOverlayRanges[nodeId][1]));
897 
898 		// Draw vertical line
899 		ctx.beginPath();
900 		ctx.moveTo(scaleX(intNodePos[0]),scaleY(intNodePos[1]));
901 		ctx.lineTo(scaleX(intNodePos[0]),scaleY(drawHeight));
902 		ctx.stroke();
903 
904 		var x1 = scaleX(intNodePos[0]) -2;
905 		var y1 = scaleY(intNodePos[1]) -2;
906 		var x2 = scaleX(intNodePos[0]) +2;
907 		var y2 = scaleY(drawHeight) + 2;
908 
909 
910 
911 		var left = scaleX(Math.min(internalNodeOverlayRanges[nodeId][0],internalNodeOverlayRanges[nodeId][1]));
912 		var width = scaleX(Math.max(internalNodeOverlayRanges[nodeId][0],internalNodeOverlayRanges[nodeId][1])) - left;
913 
914 
915 		dendV.clickRegions.addClickArea(
916 		    x1, y1,
917 		    x1, y2,
918 		    x2, y2,
919 		    x2, y1,
920 		    {
921 			nodeId: nodeId,
922 			nodeType: 'vertical',
923 			highlightBoundingbox: {
924 			    top: 0,
925 			    left: left,
926 			    width: width,
927 			    height: scaleY(drawHeight) - this.currentConfiguration.lineWidth * 2
928 			}
929 		    }
930 		);
931 
932 		p2 = intNodePos;
933 	    }
934 	}
935 
936 	// Step 3:
937 	// connect the two segments with an horizontal line
938 	// registter this line with the click array
939 	// record the position of the internal node
940 
941 	// Draw the connecting horisontal segment
942 	if(!( (typeof(p1) === 'undefined') | (typeof(p2) === 'undefined'))  ){
943 	    ctx.beginPath();
944 	    ctx.moveTo(scaleX(p1[0]), scaleY(drawHeight));
945 	    ctx.lineTo(scaleX(p2[0]), scaleY(drawHeight));
946 	    ctx.stroke();
947 
948 	    var x1 = scaleX(p1[0]) -2;
949 	    var y1 = scaleY(drawHeight ) - 2;
950 	    var x2 = scaleX(p2[0]) +2;
951 	    var y2 = scaleY(drawHeight) +2;
952 
953 	    dendV.clickRegions.addClickArea(
954 		x1, y1,
955 		x1, y2,
956 		x2, y2,
957 		x2, y1,
958 		{
959 		    nodeId: i + 1,
960 		    nodeType: 'horizontal',
961 		    highlightBoundingbox: {
962 			top: scaleY(0),
963 			left: scaleX(currentNodeLeft),
964 			width: scaleX(currentNodeRight) - scaleX(currentNodeLeft),
965 			height: scaleY(drawHeight) +  dendV.currentConfiguration.lineWidth * 2
966 		    }
967 		}
968 	    );
969 
970 	    // Keep track of the internal node position
971 	    internalNodePositions[i+1] = ([ ((p1[0]+p2[0])/2) , drawHeight ]);
972 
973 	    // Keep track of node leaf boundaries
974 	    internalNodeOverlayRanges[i+1] = [currentNodeLeft, currentNodeRight];
975 	}
976     }; // end loop over mergeArray
977 
978 
979     ctx.restore();
980 
981     this.updateCanvasCopy();
