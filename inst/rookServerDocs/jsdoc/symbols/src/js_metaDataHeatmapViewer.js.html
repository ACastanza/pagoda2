  7 /**
  8  * Manages the metadata heatmap viewer
  9  * @constructor
 10  */
 11 function metaDataHeatmapViewer() {
 12     if (typeof metaDataHeatmapViewer.instance === 'object') {
 13 	    return metaDataHeatmapViewer.instance;
 14     }
 15 
 16     // The initializer is called externally when the
 17     // dendrogram (which will provide the order)
 18     // is ready
 19 
 20     metaDataHeatmapViewer.instance = this;
 21 }

 23 /**
 24  * Initialise the metadata heatmap viewer
 25  * @description this is separate from the constructor
 26  * because it's called by the container object when things
 27  * are ready
 28  */
 29 metaDataHeatmapViewer.prototype.initialize = function () {
 30     console.log('Initializing Metadata viewer...');
 31 
 32     var metadataContainer = $('#metadata-area-container');
 33     metadataContainer.append('<div id="metadata-area-container-inner"></div>');
 34 
 35     // We need an inner container for establishing a new
 36     // reference for the absolute positioning of the two canvases
 37     var metadataContainerInner = $('#metadata-area-container-inner');
 38     metadataContainerInner.css({position: 'relative'});
 39 
 40     metadataContainerInner.append(
 41 	'<canvas id="metadata-area"></canvas>' +
 42 	'<canvas id="metadata-area-overlay"></canvas>'
 43     );
 44 
 45     var metadataArea = $('#metadata-area');
 46     metadataArea.css({
 47 	position: 'absolute',
 48 	top: 0,
 49 	left: 0
 50     });
 51 
 52     var metadataAreaOverlay = $('#metadata-area-overlay');
 53     metadataAreaOverlay.css({
 54 	position:'absolute',
 55 	top: 0,
 56 	left: 0
 57     });
 58 
 59     // Get a new clickable regions object for this heatmap
 60     this.clickRegions = new clickableRegions();
 61 
 62     // Setup the click listener
 63     // NOTE: At this point we are only interested in resoving
 64     // which line we are clicking on. In the future some of the
 65     // metadata should support clicking on individual cells and/or
 66     // regions of contigious properties.
 67     // For clicking on contigious chucks (e.g. clusters) we
 68     // will need a propery that says the the data is guaranteed to be
 69     // contigious with the default ordering
 70     (metadataAreaOverlay[0]).addEventListener('click', function(e) {
 71 	var x = e.layerX;
 72 	var y = e.layerY;
 73 
 74 	var mdhv = new metaDataHeatmapViewer();
 75 	mdhv.clickRegions.resolveClick(x,y, function(params) {
 76 	    var embV = new embeddingViewer();
 77 	    embV.setColorConfiguration('metadata');
 78 	    // Here  we are just passing the params from the
 79 	    // click region registration, we might want to change this
 80 	    // later, but in any case keep processin in here to the minimum
 81 	    embV.setMetadataColorInfo(params);
 82 	    embV.updateColors();
 83 	});
 84     });
 85 
 86     (metadataAreaOverlay[0]).addEventListener('mousemove', function(e) {
 87 	var x = e.layerX;
 88 	var metaV = new metaDataHeatmapViewer();
 89 	metaV.showOverlay(x);
 90 	var heatV = new heatmapViewer();
 91 	heatV.showOverlay(x);
 92 	var aspeV = new aspectHeatmapViewer();
 93 	aspeV.showOverlay(x);
 94     });
 95 
 96     (metadataAreaOverlay[0]).addEventListener('mouseout', function(e) {
 97 	var metaV = new metaDataHeatmapViewer();
 98 	var heatV = new heatmapViewer();
 99 
100 	metaV.clearOverlay();
101 	heatV.clearOverlay();
102 	var aspeV = new aspectHeatmapViewer();
103 	aspeV.clearOverlay()
104     });
105 
106     // Pointer change to cross hairs when over the heatmap
107      (metadataAreaOverlay[0]).addEventListener('mouseenter', function(e) {
108 	document.body.style.cursor = "crosshair";
109     });
110 
111      (metadataAreaOverlay[0]).addEventListener('mouseout', function(e) {
112 	document.body.style.cursor = "default";
113     });
114 
115 
116 
117 
118     this.updateCanvasSize();
119     this.drawMetadata();
120 }

122 /**
123  * Show an overlay vertical line
124  * @param x the x-coordinate to show the line at
125  */
126 metaDataHeatmapViewer.prototype.showOverlay = function(x) {
127     var overlayArea = document.getElementById('metadata-area-overlay');
128     var ctx = overlayArea.getContext('2d');
129 
130     var metaV = new metaDataHeatmapViewer();
131     var drawConsts = metaV.getDrawConstants();
132 
133     ctx.clearRect(0,0, overlayArea.width, overlayArea.height);
134     ctx.setLineDash([10,10]);
135     ctx.lineWidth =1 ;
136 
137     // Within the heatmap area
138     if ( x > drawConsts.left & x < drawConsts.width + drawConsts.left) {
139 	ctx.beginPath();
140 	ctx.moveTo(x, drawConsts.top);
141 	ctx.lineTo(x, drawConsts.height + drawConsts.top);
142 	ctx.stroke();
143     }
144 }

156 /**

158  * sized provided by thhe heatmapDendrogramViewer

160 metaDataHeatmapViewer.prototype.updateCanvasSize = function() {
161     console.log('metadataheatmapviewer updating canvas size');
162 
163     var metaArea =  $('#metadata-area')[0];
164     var metadataAreaOverlay = $('#metadata-area-overlay')[0];
165 
166     var metadataContainer = $('#metadata-area-container');
167 
168     heatDendV = new heatmapDendrogramViewer();
169 
170     var curWidth =  heatDendV.getCurrentWidth();
171     var curHeight =   heatDendV.getCurrentMetadataHeight();
172 
173     this.canvasElementWidth = curWidth;
174     this.canvasElementHeight =  curHeight;
175 
176     metaArea.width = curWidth;
177     metaArea.height = curHeight;
178 
179     metadataAreaOverlay.width = curWidth;
180     metadataAreaOverlay.height = curHeight;
181 
182     metadataContainer.css({width: curWidth+'px', height: curHeight+'px'});
183 }

185 /**

187  */
188 metaDataHeatmapViewer.prototype.getDrawConstants = function() {
189   var heatDendView = new heatmapDendrogramViewer();
190     var dendV = new dendrogramViewer();
191 
192     return {
193 	top: 1,
194 	left: heatDendView.getPlotAreaLeftPadding(),

196 	height: heatDendV.getCurrentMetadataHeight(),

201 /**
202  * Draw an updated metadata heatmap
203  *
204  * @todo This function should be broken up into a data prep and
205  * data plot step
206  */
207 metaDataHeatmapViewer.prototype.drawMetadata = function() {
208     // Get cells currently displayed in the dendrogram
209     var dendV = new dendrogramViewer();
210     var cellSelection = dendV.getCurrentDisplayCells();
211 
212     // Some parameters to control the  drawing of the heatmap
213     var plotConsts = this.getDrawConstants();
214 
215     var bottomPadding = 2;
216 
217     var top = plotConsts.top;
218     var left = plotConsts.left;
219     var metaWidth = plotConsts.width;
220     var metaHeight = plotConsts.height - bottomPadding;
221 
222 
223     // Request the data
224     var dataCntr = new dataController();
225     dataCntr.getCellMetadata(function(data) {
226 	// Get the canvas
227 	var canvas = $('#metadata-area')[0];
228 	var ctx =  canvas.getContext("2d");
229 
230 	// Clear the canvas
231 	// FIXME: Clear the proper area
232 	ctx.clearRect(0,0,5000,5000);
233 
234 	// Generate two full arrays with the things that were requested
235 	// One has the actual values and the other the plot colors
236 	// We will need the actual values for the rollovers later
237 
238 	// Prepare the data for plotting
239 	var renderedArrayColors = Array();
240 	var renderedArrayValues = Array();
241 	var labels = Array();
242 
243 	var j = 0;
244 	for (var key in data) {
245 	    // Skip if object property
246 	    if (! data.hasOwnProperty(key) ) continue;
247 
248 	    labels[j] = key;
249 
250 	    // TODO: Implement a hidden property
251 	    // TODO: Implement metadata row selection by the user
252 	    // TODO: Implement provided palette override by the user
253 	    // TODO: Implement default palette provision if one is not specified
254 	    // TODO: Add some futher checks --providing inconsistent data from server side will break this
255 
256 	    curEntry =  data[key];
257 
258 	    var curRow = new Array();
259 	    for (var i = 0; i < cellSelection.length; i++) {
260 		var curCell =  cellSelection[i];
261 		curRow[i] = curEntry.data[curCell];
262 	    }
263 	    renderedArrayValues[j] = curRow;
264 
265 	    // The palette from R has an alpha channel that we
266 	    // have to discard
267 	    var fixedPalette = curEntry.palette.map(function(x) {
268 		return x.substring(0,7);
269 	    });
270 
271 	    // Map to colours with provided palette
272 	    renderedArrayColors[j] = renderedArrayValues[j].map(function(x) {
273 		return (fixedPalette)[x-1];
274 	    });
275 
276 	    j++; // Output row counter
277 	} // dataCntr.getCellMetadata(
278 
279 
280 	//// Plot stage
281 	// Computed plotting param
282 	var nCells = renderedArrayColors[0].length;
283 
284 	// TODO: should be subtracting left and top here ( and
285 	//  later for the click regions as well)
286 	var cellWidth = metaWidth / (nCells);
287 	var cellHeight = metaHeight / (renderedArrayColors.length);
288 
289 	var labelYpad = cellHeight / 2 + 5;
290 	var labelXpad = 20;
291 
292 	var mdhv = new metaDataHeatmapViewer();
293 	mdhv.clickRegions.clearClickAreas();
294 
295 	for(var i = 0; i < renderedArrayColors.length; i++) {
296 	    var row = renderedArrayColors[i];
297 	    y = i * cellHeight + top;
298 	    for (var j = 0; j < row.length; j++) {
299 		    x = (j) * cellWidth + left;
300 		    ctx.fillStyle =  row[j];
301 		    ctx.fillRect(x,y, cellWidth, cellHeight);
302 	    }
303 
304 	    // Print names
305 	    var name = labels[i];
306 
307       // Cap at 16 and don't plot if smaller than 6
308 	    var fontSize = Math.min(cellHeight, 16);
309 	    if (fontSize >= 6) {
310   	    ctx.font = fontSize + 'px Arial';
311   	    ctx.fillStyle = 'black';
312   	    var labelx = (j) * cellWidth + left + labelXpad;
313   	    ctx.fillText(name, labelx , y + labelYpad);
314 	    }
315 
316 	    // Register a click region for this metadata row
317 	    var y1 = i * cellHeight + top;
318 	    var y2 = (i + 1) * cellHeight + top;
319 
320 	    var x1 = left;
321 	    var x2 = left + metaWidth;
322 
323 	    mdhv.clickRegions.addClickArea(
324 		x1, y1,
325 		x1, y2,
326 		x2, y2,
327 		x2, y1,
328 		// TODO: Is there an internal name to svae here? NOPE -- but there should be
329 		{ metadataName: labels[i] }
330 	    );
331 
332 	}
333 
334 	// Plot a bounding box
335 	ctx.beginPath();
336 	ctx.rect(left, top, metaWidth, metaHeight);
337 	ctx.stroke();
338 
339     });
340 }
