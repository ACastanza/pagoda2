  9 /**
 10  * Handles all data requests
 11  * @constructor
 12  * @description At the moment these will be simple callbacks to the backend
 13  * Rook server. In the future this will be loaded from a local
 14  * file or it can be requested from an external resource database (cross-site
 15  * scripting security issues may arise). It can also cache requests.
 16  * @param repository this is the data source currently only remote is supported this can alternatively be a local
 17  */
 18 function dataController(repository) {
 19     if (typeof dataController.instance === 'object') {
 20 	return dataController.instance;
 21     };
 22 
 23     // Init variables here
 24     this.repository = repository;
 25 
 26     // Init the cache
 27     this.cache = {};
 28     this.cache["cellmetadata"] = null;
 29     this.cache["cellorder"] = null;
 30     this.cache["reduceddendrogram"] = null;
 31     this.cache["embeddings"] = [];
 32 
 33     this.cache["lastexpressionmatrix"] = null;
 34     this.cache["lastaspectmatrix"] = null;
 35 
 36     // Define extjs objects -- used for the table data
 37     // where extjs controls are used
 38     this.defineExtJsObjects();
 39 
 40     dataController.instance = this;
 41 };

 55     var dataCntr = new dataController();

 61     } else {
 62 	$.ajax({
 63 	    dataType: "json",
 64 	    url: "getData.php",
 65 	    data: {'dataidentifier': 'hierarchy', 'type': name },
 66 	    success: function(data) {
 67 		var dataCntr = new dataController();
 68 
 69 		// The merge array values refer to a 1 indexed table
 70 		// fix this as the js tables are 0 indexed
 71 		for (var i =0; i < data.merge.length; i++) {
 72 		    // This is the case only for internal nodes
 73 		    if (data.merge[i] > 0) {
 74 			data.merge[i] = data.merge[i] - 1;
 75 		    }
 76 		}
 77 
 78 		dataCntr.cache["hierarchies"][name] = data;
 79 
 80 		callback(data);
 81 	    }

201 /**
202  * Get a dgCMatrixReader with the cell aspect weights
203  */
204 dataController.prototype.getAspectMatrix = function(cellIndexStart, cellIndexEnd, getCellNames, callback) {
205   var dataCntr = this;
206 
207   // Does the function return via callback
208   function doReturn(data) {
209 		// Unpack the data
210 		var x = dataCntr.unpackCompressedBase64Float64Array(data.x);
211 		var i = dataCntr.unpackCompressedBase64Int32Array(data.i);
212 		var p = dataCntr.unpackCompressedBase64Int32Array(data.p);
213 
214 		// This is a fix for the way R toJSON encodes one element arrays
215 		if (typeof data.Dimnames1 === 'string') {data.Dimnames1 = [ data.Dimnames1 ]; }
216 		if (typeof data.Dimnames2 === 'string') {data.Dimnames2 = [ data.Dimnames2 ]; }
217 
218 		//Convert to matrix reader and return
219 		var m = new dgCMatrixReader(i, p, data.Dim, data.Dimnames1,
220 					    data.Dimnames2, x);
221 		callback(m);
222   }
223 
224   // Check input
225   if (!Number.isInteger(cellIndexStart)) { throw new Error("cellIndexStart must be an integer"); }
226 	if (!Number.isInteger(cellIndexEnd)) { throw new Error("cellIndexEnd must be an interger"); }
227 
228 
229 	// Check the cache
230 	if(this.cache.lastaspectmatrix !== null ) {
231 	    if ( this.cache.lastaspectmatrix.cellindexstart === cellIndexStart &
232   		 this.cache.lastaspectmatrix.cellindexend === cellIndexEnd &
233   		 this.cache.lastaspectmatrix.getCellNames === getCellNames) {
234     		doReturn(this.cache.lastaspectmatrix.data);
235     		return null; // No ajax request to return
236 	    }
237 	}
238 
239 
240   // Setup the request data
241 	var requestData = {
242 	    "dataidentifier": "aspectmatrixsparsebyindexbinary",
243 	    "cellindexstart": cellIndexStart,
244 	    "cellindexend": cellIndexEnd,
245 	    "getcellnames": getCellNames
246 	};
247 
248 
249 	var ajaxRequest = $.ajax({
250 	    type: "GET",
251 	    dataType: "json",
252 	    url: "getData.php",
253 	    data: requestData,
254 	    success: function(data) {
255 
256 		// Check if the returned data are valid
257 		if (typeof data !== 'object') {
258 		    throw new Error('Returned data is not of type object');
259 		}
260 		if (! data.hasOwnProperty('i') ){
261 		    throw new Error('data object does not have an i field');
262 		}
263 		if (! data.hasOwnProperty('p')) {
264 		    throw new Error('data object does not have a p field');
265 		}
266 		if (! data.hasOwnProperty('Dim')) {
267 		    throw new Error('data object does not have a Dim field');
268 		}
269 		if (! data.hasOwnProperty('Dimnames1')) {
270 		    throw new Error('data object does not have Dimnames1 field');
271 		}
272 		if (! data.hasOwnProperty('Dimnames2')){
273 		    throw new Error('data object does not have Dimnames2 field');
274 		}
275 		if (! data.hasOwnProperty('x')) {
276 		    throw new Error('data object does not have x field');
277 		}
278 
279 		// Update the cache
280 		dataCntr.cache.lastaspectmatrix = {};
281 		dataCntr.cache.lastaspectmatrix.cellindexstart = cellIndexStart;
282 		dataCntr.cache.lastaspectmatrix.cellindexend = cellIndexEnd;
283 		dataCntr.cache.lastaspectmatrix.getCellNames = getCellNames;
284 		dataCntr.cache.lastaspectmatrix.data = data;
285 
286 		doReturn(data);
287 	    }
288 	});
289 
290 	return ajaxRequest;
291 }

501 /**
502  * Given a string that encodes a sequence of lz compressed
503  * Int32 values, decompress and return resulting typed array
504  * @param encoded the encoded array
505  * @return Int32Array with decoded values
506  */
507 dataController.prototype.unpackCompressedBase64Int32Array = function(encoded) {
508     var compressed = atob(encoded);
509     var raw = pako.inflate(compressed);
510     var int32aBuffer = new Uint8Array(raw);
511     var int32a = new Int32Array(int32aBuffer.buffer);
512 
513     return int32a;
514 }

541 dataController.prototype.getAvailableEmbeddings =  function(type, callback, callbackParams ) {
542     $.ajax({
543 	dataType: "json",
544 	url: "getData.php",
545 	data: {'dataidentifier': 'availableembeddings', 'type': type},
546 	success: function(data) {
547 	    // The returned data can be:
548 	    //  * a string
549 	    //  * an array of strings
550 	    //  * null
551 
552 	    if (!( data === null | typeof data === 'string' | Array.isArray(data) )) {
553 		throw new Error('data needs to be one of null, string or array');
554 	    }
555 	    // If an array check that it is not empty and that all elements are strings
556 	    if (Array.isArray(data)) {
557 		if(data.length === 0) {
558 		    throw new Error('data is an array of length 0');
559 		}
560 		for (var i = 0; i < data.length; i++) {
561 		    if (typeof data[i] !== 'string') {
562 			throw new Error('Element ' + i + 'of data is not a string');
563 		    }
564 		}
565 	    }
566 
567 	    // NOTE if only one value is returned the rjson library will
568 	    // just return the string literal and this will cause problems later
569 	    // Checking and fixing:
570 	    if (typeof data === "string") {
571 	       data = [data];
572 	    }
573 
574 	    callback(data, callbackParams);
575 	}
576     });
577 };

689 		for ( var key in data ) {

691 			throw new Error('data member of the metadata entry ' + key + ' is not an object');
692 		    }
693 		    if ( data[key]['data'].length === 0) {
694 			throw new Error('data member of the metadata entry ' + key + ' is empty');
695 		    }
696 
697 
698 		    if ( typeof data[key]['palette'] !== 'object' ){
699 			throw new Error('palette member of the metadata entry ' + key + ' is not an object');
700 		    }
701 		    if (data[key]['palette'].length === 0 ) {
702 			throw new Error('palette member of the metadata entry ' + key + ' is empty');
703 		    }
704 
705 		}
706 
707 		// TODO: Check all meta data contain the same number of entries
708 		// TODO: Check that all the values have a palette match
709 
710 		dataCntr.cache["cellmetadata"] = data;
711 		callback(data);
712 	    }
713 	});
714     }
