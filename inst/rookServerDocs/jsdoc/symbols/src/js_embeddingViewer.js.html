  3 /**
  4  * Responsible for for handling the embedding panel
  5  * This class does not actually do any plotting. It delegates the plotting
  6  * to a suitable class depending on the type of the embedding. It does 
  7  * however maintain the plot state, the individual embedders should not
  8  * maintain state
  9  * Singleton
 10  * @constructor
 11  */
 12 function embeddingViewer() {
 13     if (typeof embeddingViewer.instance === 'object') {
 14 	return embeddingViewer.instance;
 15     };
 16 
 17     console.log('Initialising embeddingViewer...');
 18 
 19     // Singleton
 20     embeddingViewer.instance = this;
 21 
 22     // FIXME: This needs to be done before the event listeners are configured
 23     // and after the instance is set to this. This is a race condition 
 24     // for the backend
 25     this.generateToolbar();
 26 
 27     this.loadDefaultConfig();
 28 
 29     // Generate and show a viewer, save it in the current viewer variable
 30     // so we can send commands to it later
 31 
 32     // NOTE: This is the original d3 viewer with performace issues
 33     //this.currentViewer = new scatterEmbeddingD3();
 34 
 35 
 36     // If we ever implement multiple viewers the appropriate containers
 37     // need to be cleanedup
 38     // This is the currentViewer that plots with canvas
 39     this.currentViewer = new embeddingViewerScatterCanvas();
 40 
 41 
 42     this.generateWait();
 43 }; // function embeddingViewer() {

 47 /**
 48  * Shows the wait overlay
 49  */
 50 embeddingViewer.prototype.showWait = function() {
 51     
 52     var extJsContainer = Ext.getCmp('embeddingExtJSWrapper');
 53     var plotHeight = extJsContainer.body.getHeight(true);
 54     var plotWidth = extJsContainer.body.getWidth(true);
 55 
 56     var embWaitDiv = $('#embedding-wait');
 57     
 58     var top = (plotHeight - embWaitDiv.height() ) /2;
 59     var left = (plotWidth - embWaitDiv.width() ) /2;
 60 
 61     embWaitDiv.css({
 62 	'visibility': 'visible',
 63 	'top': top,
 64 	'left': left
 65     });
 66 }

 99 /**
100  * Generate the toolbar for the embedding viewer and populate
101  * it with options from the dataController
102  */
103 embeddingViewer.prototype.generateToolbar = function() {
104 
105     // Define a store for the options
106     var embeddingOptionsStore = Ext.create('Ext.data.Store', {
107 	fields: ['reduction', 'embedding', 'label', 'value'],
108 	id: 'embeddingOptionsStore'
109     });
110 
111     // Make a combobox
112     var comboBox = Ext.create('Ext.form.ComboBox', {
113 	fieldLabel: 'Select Embedding',
114 	store: embeddingOptionsStore,
115 	queryMode: 'local',
116 	displayField: 'label',
117 	editable: false,
118 	valueField: 'value'
119     });
120 
121     // Set the option change listener
122     comboBox.addListener('change', function(e) { 
123 	var value =  e.value;
124 	var embeddingIds = value.split(':');
125 	embView = new embeddingViewer();
126 	embView.showEmbedding(embeddingIds[0],embeddingIds[1]);
127     });
128 
129     // Menu for embedding settings
130     var embeddingSettingsMenu = Ext.create('Ext.menu.Menu', {
131 	id: 'embeddingSettingsMenu',
132 	items: [{
133 	    fieldLabel: 'Point size',
134 	    xtype: 'numberfield',
135 	    tooltip: 'Plot point size',
136 	    value: p2globalParams.embedding.defaultPlotPointSize, // 2
137 	    minValue: 0.1,
138 	    maxValue: 20,
139 	    disabled: false,
140 	    listeners: {
141 		change: {buffer: 800, fn: function(f,v){
142 		    var embViewer = new embeddingViewer();
143 		    embViewer.setCurrentPointSize(v);
144 		    embViewer.redraw();
145 		}} // change buffered listerne
146 	    }// listeners
147 	}, 
148 	{
149 	    fieldLabel: 'Opacity',
150 	    xtype: 'numberfield',
151 	    tooltip: 'Plot point opacity',
152 	    value: p2globalParams.embedding.defaultAlpha,
153 	    minValue: 0.0001,
154 	    maxValue: 1.0,
155 	    step: 0.1,
156 	    disabled: false,
157 	    listeners: {
158 		change: {buffer: 800, fn: function(f,v){
159 		    var embViewer =  new embeddingViewer();
160 		    embViewer.setCurrentAlpha(v);
161 		    embViewer.redraw();
162 		}} // change
163 	    } // listeners
164 	},
165 	{
166 	    text: 'Show border',
167 	    checked: p2globalParams.embedding.defaultBorder,
168 	    checkHandler: function(f, v){ 
169 		var embViewer = new embeddingViewer();
170 		embViewer.setCurrentBorder(v);
171 		embViewer.redraw();
172 
173 	    }
174 	},
175 	{
176 	    fieldLabel: 'Border width',
177 	    xtype: 'numberfield',
178 	    tooltip: 'Plot point border width',
179 	    value: p2globalParams.embedding.defaultBorderWidth,
180 	    minValue: 0.001,
181 	    maxValue: 3,
182 	    step: 0.25,
183 	    disabled: false,
184 	    listeners: {
185 		change: {buffer: 800, fn: function(f, v){
186 		    var embViewer = new embeddingViewer();
187 		    embViewer.setCurrentBorderWidth(v);
188 		    embViewer.redraw();
189 		}} // change
190 	    }    
191 	}
192 	], // items
193     });
194 
195     // Add it to a toolbar in the appropriatepanel
196     var embeddingPanel = Ext.getCmp('embeddingExtJSWrapper');
197     var embeddingPanelHeader = embeddingPanel.getHeader();
198     var toolbar = Ext.create('Ext.Toolbar');
199     toolbar.add(comboBox);
200     toolbar.add({xtype: 'tbseparator'});
201     // Add a button for the settings menu
202     toolbar.add(
203 	{
204 	    text: "Settings",
205 	    type: "button",
206 	    tooltip: "Customize embedding view",
207 	    glyph: 0xf013,
208 	    menu: embeddingSettingsMenu
209 	}
210 	
211     );
212 
213     embeddingPanelHeader.add(toolbar);
214 
215 
216     // Delay populating the menu above as this will not be used 
217     // as soon as the app loads and it helps reduce the inital swarm of requests
218     setTimeout(this.populateMenu, 2000)
219 
220 }; // generateToolbar

222 /**
223  * Populates the available embeddings menu
224  * @private
225  * @todo this is slow, and generates many requests we should remodel 
226  * the data controller to get all the data that we want with 
227  * one request, also account for the fact that some embeddings 
228  * might not correspond to reduction
229  * Also embedding will now need to have a plot type
230  */
231 embeddingViewer.prototype.populateMenu = function() {
232     // Populate the menu using the data controller -- async
233     var datCntr = new dataController();
234     datCntr.getAvailableReductionTypes(function(reductions){
235 	for ( var i = 0; i < reductions.length; i++ ) {
236 	    var currReduction = reductions[i];
237 
238 
239 	    datCntr.getAvailableEmbeddings(currReduction, function(embeddings, currReduction) {
240 
241 		embeddingOptionstore =  Ext.data.StoreManager.lookup('embeddingOptionsStore');
242 
243 		if (embeddings !== null) {
244 
245  		    for (var j = 0; j < embeddings.length; j++) {
246 			var embeddingIdentifier =  currReduction + ':' + embeddings[j];
247 			var embeddingLabel =  currReduction + ' --> ' + embeddings[j];
248 
249 			if (currReduction === p2globalParams.embedding.defaultEmbedding.reduction &&
250 			    embeddings[j] === p2globalParams.embedding.defaultEmbedding.embedding) {
251 			    // The default
252 			}
253 
254 			embeddingOptionstore.add({ 
255 			    'reduction': currReduction, 
256 			    'embedding':  embeddings[j], 
257 			    "label": embeddingLabel, 
258 			    "value": embeddingIdentifier
259 			});
260 
261 		    } // for 
262 
263 		}
264 	    }, currReduction // For the callback extra information
265 					  ) // getAvailableEmbeddings
266 	} //for 
267     }); // getAvailableReductionTypes
268 }
