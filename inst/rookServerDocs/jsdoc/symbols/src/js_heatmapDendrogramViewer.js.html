 53 /**
 54  * Obtain the size of the container element from ExtJS and store it
 55  * in the object so that the children can access it
 56  */
 57 heatmapDendrogramViewer.prototype.updateContainerSize = function() {
 58     var extJsContainer = Ext.getCmp('mainViewPanel');
 59     // TODO: move the -30 to some centralised configration
 60     this.viewerHeight = extJsContainer.body.getHeight(true) - 30 ;
 61     this.viewerWidth = extJsContainer.body.getWidth(true);
 62 }

 64 /**
 65  * Get the element skipping value
 66  * @param n {number} number of elements to plot
 67  * @description Get the element skipping value given the
 68  * number of elements we are plotting. We keep this in this object
 69  * so that it exists in a centralised place and the skipping is
 70  * consistent between different components. Currently not used
 71  * @todo Improve this
 72  */
 73 heatmapDendrogramViewer.prototype.getSkipping = function(n) {
 74     var k =  1000;
 75     var p = 4 ;
 76 
 77     var skipping;
 78     if ( n < k) {
 79 	skipping = 1;
 80     } else {
 81 	skipping = p;
 82     }
 83     return skipping;
 84 };

 86 /**
 87  * Get the current height of the canvas area for the dendrogram.
 88  * @description get the current height of the canvas area for the dendrogram
 89  * @returns current canvas height for the dendrogram. This is currently fixed
 90  */
 91 heatmapDendrogramViewer.prototype.getCurrentDendHeight = function() {

190 /**
191  * Call the component initializers and do a view update
192  */
193 heatmapDendrogramViewer.prototype.initializeComponents = function() {
194   // Initialize heatmap and meta after dendrogram
195   var evtBus = new eventBus();
196   var dendrogramInitListener = function() {
197     var heatView = new heatmapViewer();
198     heatView.initialize();
199 
200     var metaView = new metaDataHeatmapViewer();
201     metaView.initialize();
202 
203     var aspectHeatView = new aspectHeatmapViewer();
204     aspectHeatView.initialize();
205 
206     // Unregister this handler
207     var evtBus = new eventBus();
208     evtBus.unregister("dendrogram-cell-order-updated", null, dendrogramInitListener);
209   }
210   evtBus.register("dendrogram-cell-order-updated", null,dendrogramInitListener);
211 
212   var dendView = new dendrogramViewer();
213   dendView.initialize();
214 }

217 /**
218  * Do a view update given the current object settings
219  */
220 heatmapDendrogramViewer.prototype.updateView = function() {
221     // Initialize heatmap after dendrogram
222     var evtBus = new eventBus();
223 
224     // Use a named function so we can remove it from the listeners
225     var dendrogramInitListener = function() {
226 	var heatView = new heatmapViewer();
227 	var metaView = new metaDataHeatmapViewer();
228 	var aspectHeatView = new aspectHeatmapViewer();
229 
230 	heatView.updateCanvasSize();
231 	heatView.drawHeatmap();
232 
233 	metaView.updateCanvasSize();
234 	metaView.drawMetadata();
235 
236 	aspectHeatView.updateCanvasSize();
237 	aspectHeatView.drawHeatmap();
238 
239 	// Unregister this handler
240 	var evtBus = new eventBus();
241 	evtBus.unregister("dendrogram-cell-order-updated", null, dendrogramInitListener);
242   }
243 
244     // Listen for first dendrogram initialization
245     evtBus.register("dendrogram-cell-order-updated", null,dendrogramInitListener);
246 
247     var dendView = new dendrogramViewer();
248     dendView.redrawDendrogram();
249 //    dendView.updateOverlay();
250 };
