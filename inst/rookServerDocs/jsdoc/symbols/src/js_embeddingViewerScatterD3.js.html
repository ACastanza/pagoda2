 14 function scatterEmbeddingD3() {
 15     console.log('Initialising scatterEmbeddingD3');
 16 
 17     // Here we need to tap into the extjs container
 18     // component resize function 
 19     var element = Ext.getCmp('embedding-app-container');
 20     element.onResize = function(){
 21 	embView = new embeddingViewer();
 22 	embView.redraw();
 23     };
 24 
 25 
 26     // TODO: Move this to the main embedding drawing object
 27     var evtBus = new eventBus();
 28     // Dendrogram node has been clicked
 29     // Set the coloring scheme to dendrogram ( which uses the selection controller )
 30     // and call updateColors
 31     evtBus.register("dendrogram-node-click", null, function(e,p) {
 32 	var embV = new embeddingViewer();
 33 
 34 	// Update  configuration of embedding
 35 	// The actual selected node is passed by the cell selection
 36 	embV.setColorConfiguration('dendrogram');
 37 	embV.setDendrogramColorInfo({nodeType: p.type});
 38 
 39 	embV.updateColors();
 40     }); 
 41 }

 47 /**
 48  * Update the embedding colors to match metadata information
 49  */
 50 scatterEmbeddingD3.prototype.updateColorsMetadata = function() {
 51     // Get the configuration information that we need
 52     var ev = new embeddingViewer();
 53     var config = ev.getConfig();
 54 
 55 
 56     var dataCntr = new dataController();
 57 
 58     // Get the embedding viewer config and the name
 59     // of the metadata we want to plot
 60     var metadataColorInfo = config.metaDataColorInfo;
 61     var metadataName = metadataColorInfo.metadataName;
 62 
 63     // FIXME: cellSelection on the data controller for the metadata
 64     // doesn't work so we are passing null here. We want to move
 65     // towards a model where we don't specify cells at all if we 
 66     // don't want anything other than everything in the default order
 67     // TODO: We would like to minimize movement of data 
 68     // so we should have the ability to specify which metadata we want
 69     // and  only get that 
 70     dataCntr.getCellMetadata(function(data) {
 71 	colorData = data[metadataName].data;
 72 	colorPalette = data[metadataName].palette;
 73 
 74 	// Prep data for d3
 75 	var d3data =[];
 76 	for (i in colorData) {
 77 	    var clusterId = Number(colorData[i]);
 78 	    var clusterColor = colorPalette[clusterId-1];
 79 	    // Remove the alpha channel
 80 	    clusterColor = clusterColor.substring(0,7);
 81 	    d3data[d3data.length] = [i, clusterId ,clusterColor];
 82 	 }
 83 
 84 	// Update the colors
 85 	var embV = new embeddingViewer();
 86 	var alpha = embV.getCurrentAlpha();
 87 
 88 	d3.select('#embedding-draw')
 89 	    .selectAll('circle')
 90 	    .data(d3data)
 91 	    .attr('opacity', alpha)
 92 	    .attr('fill', function(d) {
 93 		return d[2];
 94 	    });
 95     });
 96 }

 98 scatterEmbeddingD3.prototype.plotEmbedding = function() {
 99     var dataCntr = new dataController();
100 
101     // Get the configuration information that we need
102     var ev = new embeddingViewer();
103     var config = ev.getConfig();
104 
105     var type = config["type"];
106     var embeddingType = config["embeddingType"];
107 
108     dataCntr.getEmbedding(type,embeddingType, function(data){
109 	// TODO: embeddings must have type to implement things other
110 	// than scatter plots, this is best implemented as separate classes
111 	// e.g. scatterPlotEmbedder with some kind of virtual function mechanism
112 
113 	plotData = pagHelpers.jsonSerialisedToArrayOfArrays(data);
114 
115 	// Clean-up existing
116 	var embeddingContainer = $("#embedding-draw");
117 	embeddingContainer.empty();
118 
119 	// TODO: Check if the event handlers need cleaning - potential mem leak
120 	var extJsContainer = Ext.getCmp('embeddingExtJSWrapper');
121 	var plotHeight = extJsContainer.body.getHeight(true);
122 	var plotWidth = extJsContainer.body.getWidth(true);
123 
124 	// Which dim is limiting?
125 	size = Math.min(plotWidth, plotHeight);
126 
127 	var svg = d3.select("#embedding-draw")
128 	    .append("svg")
129 	    .attr("width", size)
130 	    .attr("height", size);
131 
132 	// Center it in x and y
133 	var vpad = 0;
134 	var hpad = 0;
135 
136 	if (plotHeight < plotWidth) {
137 	    hpad = (plotWidth - size)/2
138 	} else {
139 	    vpad = (plotHeight - size) /2
140 	}
141 	
142 	// For getting the plot size
143 	var embViewer =  new embeddingViewer();
144 
145 	$('#embedding-draw > svg').css('margin-top', vpad);
146 	$('#embedding-draw > svg').css('margin-left', hpad);
147 
148 	// Slight shrink the window scale to avoid plotting on the edges
149 	var rangeScaleFactor = 0.95;
150 
151 	var xScale =  d3.scaleLinear()
152 	    .domain([d3.min(plotData, function(d) { return d[1]; }), d3.max(plotData, function(d) { return d[1]; })])
153 	    .range([(size * (1 - rangeScaleFactor)), size * rangeScaleFactor]);
154 
155 	var yScale = d3.scaleLinear()
156 	    .domain([d3.min(plotData, function(d) { return d[2]; }), d3.max(plotData, function(d) { return d[2]; })])
157 	    .range([(size * (1 - rangeScaleFactor)), size * rangeScaleFactor]);
158 
159 	// The size to plot at
160 	var pointsize = embViewer.getCurrentPointSize();
161 	var alpha = embViewer.getCurrentAlpha();
162 	
163 	// The border
164 	var stroke = '';
165 	var strokeWidth = '1';
166 	if (embViewer.getCurrentBorder() === true ) {
167 	    stroke = embViewer.getCurrentBorderColor();
168 	    strokeWidth = embViewer.getCurrentBorderWidth();
169 	}
170 
171 	// Plot the datapoints
172 	svg.selectAll("circle")
173 	    .data(plotData)
174 	    .enter()
175 	    .append("circle")
176 	    .attr("cx", function(d){return xScale(d[1]) })
177 	    .attr("cy",function(d){return yScale(d[2]) })
178 	    .attr("r", pointsize)
179 	    .attr("opacity", alpha)
180 	    .attr("stroke", stroke)
181 	    .attr("stroke-width", strokeWidth)
182 	    .attr("id", function(d) { return "plotpoint_" + d[0] })
183 	    .on("mouseover",  function() {
184 		// Generate event on eventBus
185 		var eB = new eventBus();
186 		
187 		// Remove the d3 point id prefix
188 		var re = new RegExp('^plotpoint_');
189 		var cellId = this.id;
190 		cellId.replace(re, '');
191 
192 		eB.publish("embedding-mouseover-plotpoint", {cellId: this.id});
193 		
194 	    })
195 	    .on("click", function() {
196 		// Generate event on eventBus
197 		var eB =  new eventBus();
198 		eB.publish("embedding-click-plotpoint", {cellId: this.id});
199 	    });
200     });
201 
202     // Update the colors to what is currently selected
203     this.updateColors();
204 } // embeddingViewer.prototype.plotEmbedding

206 /**  
207  * Update the colors -- this embedder does not need to redraw
208  */
209 scatterEmbeddingD3.prototype.updateColors = function() {
210     var ev = new embeddingViewer();
211     var config = ev.getConfig();
212 
213     var colconfig = config.colors;
214 
215     if (typeof colconfig === 'undefined') return;
216 
217     if( colconfig  === "dendrogram") {
218 	this.updateColorsDendrogram();
219     } else if ( colconfig === 'metadata') {
220 	this.updateColorsMetadata();
221     } else if ( colconfig === 'geneexpression') {
222 	this.updateColorsGeneexpression();
223     } else {
224 	throw new Error('Unknown embedding color setting: ' + colconfig );
225     }
226 }

236 /**
237  * Update the colors using dendrogram information 
238  */
239 scatterEmbeddingD3.prototype.updateColorsDendrogram = function() {
240         var embV = new embeddingViewer();
241         var dendV = new dendrogramViewer();
242 
243 	if (embV.currentConfiguration.dendrogramColorInfo.nodeType == "vertical") {
244 	    var dataCntr = new dataController();
245 	    var selCntr =  new cellSelectionController();
246 	    var embV = new embeddingViewer();
247 	    var type= embV.currentConfiguration["type"];
248 	    var embeddingType =  embV.currentConfiguration["embeddingType"];
249 	    var curPrimSel = selCntr.getSelection("currentPrimarySel");
250 
251 	    dataCntr.getEmbedding(type, embeddingType, function(data) {
252 		plotData = pagHelpers.jsonSerialisedToArrayOfArrays(data);
253 		// Amend the data according to the cell selection
254 		for (var i = 0; i < plotData.length; i++) {
255 		    var group;
256 
257 		    if ($.inArray(plotData[i][0], curPrimSel) > -1) {
258 			plotData[i][3] = "selected";
259 		    } else {
260 			plotData[i][3] = "notselected";
261 		    }
262 		}
263 
264 		var alpha = embV.getCurrentAlpha();
265 		// Update d3
266 		d3.select("#embedding-draw")
267 		    .selectAll("circle")
268 		    .data(plotData)
269 		    .attr("opacity", alpha)
270 		    .attr("fill", function(d){ 
271 			if (d[3] === "selected") {
272 			    return p2globalParams.dendrogram.selectionColors.selectedMain;
273 			};
274 			return p2globalParams.dendrogram.selectionColors.deselected;
275 		    });
276 	    });
277 	} else if (embV.currentConfiguration.dendrogramColorInfo.nodeType == "horizontal") {
278 	    var dataCntr = new dataController();
279 	    var selCntr =  new cellSelectionController();
280 	    var embV = new embeddingViewer();
281 	    var type= embV.currentConfiguration["type"];
282 	    var embeddingType =  embV.currentConfiguration["embeddingType"];
283 
284 	    dataCntr.getEmbedding(type, embeddingType, function(data) {
285 		plotData = pagHelpers.jsonSerialisedToArrayOfArrays(data);
286 
287 		    var curPrimSel = selCntr.getSelection("currentPrimarySel");
288 		    var curAltSel = selCntr.getSelection("currentAltSel");
289 
290 		// Amend the data according to the cell selection
291 		for (var i = 0; i < plotData.length; i++) {
292 		    var group;
293 
294 		    if ($.inArray(plotData[i][0], curPrimSel) > -1) {
295 			plotData[i][3] = "left";
296 		    } else if ($.inArray(plotData[i][0], curAltSel) > -1) {
297 			plotData[i][3] = "right";
298 		    } else {
299 			plotData[i][3] = "notselected";
300 		    }
301 		}
302 
303 		var alpha = embV.getCurrentAlpha();
304 		d3.select("#embedding-draw")
305 		    .selectAll("circle")
306 		    .data(plotData)
307 		    .attr("opacity", alpha)
308 		    .attr("fill", function(d){ 
309 			if (d[3] === "left") {
310 			    return p2globalParams.dendrogram.selectionColors.selectedMain;
311 			} else if (d[3] === "right") {
312 			    return p2globalParams.dendrogram.selectionColors.selectedAlt;
313 			}
314 			return p2globalParams.dendrogram.selectionColors.deselected;
315 		    }); // d3.select
316 	    });
317 	} else { 
318 	    console.error("Unknown click type");
319         }; // if (p.type == "vertical")
320 }
