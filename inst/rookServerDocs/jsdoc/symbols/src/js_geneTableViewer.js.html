1 "use strict";
  2 /**
  3  * Responsible for handling the gene
  4  * table creation and any required updates. Singlenton
  5  * @constructor
  6  */
  7 function geneTableViewer()  {
  8     if (typeof geneTableViewer.instance === 'object') {
  9 	return geneTableViewer.instance;
 10     }
 11 
 12     console.log('Initializing geneTableViewer...');
 13 
 14     this.generateGeneTable();
 15 
 16     geneTableViewer.instance = this;
 17 };
 18 
 19 
 20 /**
 21  * Generate the gene table
 22  * Here were are making the gene table and connecting it 
 23  * to the appropriate data source provided by the dataController
 24  */
 25 geneTableViewer.prototype.generateGeneTable = function() {
 26     var dataCntr = new dataController();
 27     dataCntr.getGeneInformationStore(function(geneTableEntryStore) {
 28 
 29 	// For checkboxes on table
 30 	var geneTableSelectionModel =  Ext.create('Ext.selection.CheckboxModel', {});
 31 
 32 	// Construct the table
 33 	var geneTable = Ext.getCmp('geneTableViewerExtJS');
 34 	geneTable.add(Ext.create('Ext.grid.Panel',{
 35 	    title: '',
 36 	    id: 'extjsgenetable',
 37 	    selModel: geneTableSelectionModel,
 38 	    emptyText: 'No matching genes',
 39 	    store: geneTableEntryStore,
 40 	    columns: [
 41 		{ text: 'Name', dataIndex: 'genename', width: '80%',
 42 		  // Custom render for links of gene names
 43 		  renderer: function(value) {
 44 		      return Ext.String.format(p2globalParams.misc.jaxGeneQueryFormatString,value,value)
 45 		  },
 46 		},
 47 		{ text: 'Dispersion', dataIndex: 'dispersion', width: '20%' },
 48 //		{ text: 'Score', dataIndex: 'score', width: '20%' }
 49 	    ], // columns
 50 	    height: '100%',
 51 	    width: '100%',
 52 	    singleSelect: false,
 53 	    listeners: {
 54 		// Hide the refresh button
 55 		afterrender: function() {this.down('#refresh').hide();},
 56 	    }, // listeners
 57 
 58 	    tbar: Ext.create('Ext.PagingToolbar', {
 59 		//store: geneTableEntryStore,
 60 		displayInfo: true,
 61 		prependButtons: true,
 62 
 63 		items: [
 64 		    {
 65 			emptyText: 'Search...',
 66 			xtype: 'textfield',
 67 			width: 100,
 68 			listeners: {
 69 			    'change': {buffer: 50, fn: function(f, newValue, oldValues, eOpts) {
 70 				var g = Ext.getCmp('extjsgenetable');
 71 				var store = g.getStore();
 72 				store.clearFilter();
 73 				if (newValue !== '') {
 74 				    store.filterBy(function(rec) {
 75 					if (rec.get('genename').match(new RegExp(newValue,'i'))) {
 76 					    return true;
 77 					} else {
 78 					    return false;
 79 					} // if genename
 80 				    }); // store filter by
 81 				} // if new values
 82 			    }} //change listener and buffer
 83 			} // listeners
 84 		    },
 85 		    {
 86 			type: "button", 
 87 			text: 'Show selected',
 88 			tooltip: 'Show selected genes in main heatmap',
 89 			glyph: 0xf0ce,
 90 			handler: function() {
 91 			    var heatmapV = new heatmapViewer();
 92 			    heatmapV.setNamedSelectionToDisplayGenes('geneTableSelection');
 93 			    heatmapV.drawHeatmap();
 94 			} //handler
 95 		    }, //button
 96 		    {xtype: 'tbseparator'},
 97 
 98 		]
 99 	    }), //tbar
100 	    listeners: {
101 		'selectionchange': function(selected, eOpts) {
102 		    var selectedGeneNames =  [];
103 
104 		    var selectedItems = selected.getSelected();
105 		    selectedItems.each(function(item,index,length){
106 			selectedGeneNames.push(item.data.genename);
107 		    });
108 		    
109 		    var geneSelCntr =  new geneSelectionController();
110 		    geneSelCntr.setSelection('geneTableSelection', selectedGeneNames);
111 		}
112 
113 	    } // listeners
114 	}));
115 
116     });
117 
118 }
