 24 /**
 25  * Initialise the application components
 26  */
 27 function initialise() {
 28     // Generate the overall layout
 29     generateExtJsLayout();
 30 
 31     // Initialize internal components
 32     var dataCntr = new dataController('remote');
 33 
 34     // Calculation controllers are init from a factory that is singleton
 35     var calcCntr = new calculationControllerFactory('local');
 36 
 37     var evtBus = new eventBus();
 38     var stsBar = new statusBar();
 39     var selCntr = new cellSelectionController();
 40     var infoBxCntr = new infoboxController();
 41     var geneSelCntr = new geneSelectionController();
 42 
 43     // Controller for cell selection UI
 44     var cellSelUICntr = new cellSelectionUIcontroller();
 45     var geneSelUICntr = new geneSelectionUIcontroller();
 46     var actionPanelUICntr = new actionPanelUIcontroller();
 47 
 48     // Set the page title
 49     document.title = p2globalParams.generalParams.applicationName;
 50 
 51     // Initialize page components
 52     embView = new embeddingViewer();
 53     // Load the default embedding
 54     embView.showEmbedding(p2globalParams.embedding.defaultEmbedding.reduction,
 55 			  p2globalParams.embedding.defaultEmbedding.embedding);
 56 
 57 
 58     // Generate the tables
 59     var geneTable = new geneTableViewer();
 60 //  var odGeneTable = new odGeneTableViewer();
 61     var geneSetsTable = new geneSetsTableViewer();
 62     var heatDendView = new heatmapDendrogramViewer();
 63     var aspTableView = new aspectsTableViewer();
 64 
 65 
 66     // Update status bar
 67     stsBar.showMessage("Ready");
 68 };
