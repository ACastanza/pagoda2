  7 /**
  8  * Object that manages the aspect heatmap viewer
  9  */
 10 function aspectHeatmapViewer() {
 11   if (typeof aspectHeatmapViewer.instance === 'object') {
 12     return aspectHeatmapViewer.instance;
 13   }
 14   console.log('Initializing aspect heatmap viewer...');
 15 
 16   aspectHeatmapViewer.instance = this;
 17 }

 20 /**
 21  * Perform initialization of the aspect heatmap viewer
 22  * @description This is called by the parent heatmapDendrogram object
 23  * when initialization has finished.
 24  */
 25 aspectHeatmapViewer.prototype.initialize = function() {
 26 
 27   // Clickable regions for cross hairs and navigating
 28   this.aspectRegions = new clickableRegions();
 29 
 30   var aspectHeatmapContainer = $('#aspect-heatmap-container');
 31   aspectHeatmapContainer.css({position: 'relative'});
 32 
 33   aspectHeatmapContainer.append(
 34     '<canvas id="aspect-heatmap-area"></canvas>' +
 35     '<canvas id="aspect-heatmap-area-overlay"></canvas>'
 36   );
 37 
 38   var aspectHeatmapArea = $('#aspect-heatmap-area');
 39   aspectHeatmapArea.css({
 40     position: 'absolute',
 41     top: 0,
 42     left: 0
 43   });
 44 
 45   var aspectHeatmapAreaOverlay = $('#aspect-heatmap-area-overlay');
 46   aspectHeatmapAreaOverlay.css({
 47     position: 'absolute',
 48     top: 0,
 49     left: 0
 50   });
 51 
 52   // Setup events for overlays
 53   this.setupOverlays();
 54 
 55   // update the size of both canvases
 56   this.updateCanvasSize();
 57 
 58   // TODO: Update this with an apsect specific palette
 59       this.palManager = new paletteManager();
 60     this.palManager.setPalette(p2globalParams.heatmapViewer.defaultPaletteName);
 61     this.palManager.setNumberOfColors(p2globalParams.heatmapViewer.defaultPaletteLevels);
 62 
 63   // Draw the heatmap
 64   this.drawHeatmap();
 65 
 66 };

 68 /**
 69  * Setup the event listeners for the overlay effects
 70  */
 71 aspectHeatmapViewer.prototype.setupOverlays = function() {
 72   var heatmapOverlayArea = $('#aspect-heatmap-area-overlay')[0];
 73   var aspHeatView = this;
 74 
 75   heatmapOverlayArea.addEventListener('click', function(e) {
 76     var x = e.layerX;
 77     var y = e.layerY;
 78 
 79     var regionData = aspHeatView.aspectRegions.resolveClick(x,y);
 80     if (typeof regionData !== 'undefined') {
 81 
 82       var aspTable = new aspectsTableViewer();
 83       aspTable.showSelectedAspect(regionData.aspectId);
 84 
 85       // TODO: Uncomment when embedding viewer supports this
 86       /*
 87       var embV = new embeddingViewer();
 88 	    embV.setColorConfiguration('aspect');
 89 	    embV.setAspectColorInfo({aspectid: regionData.aspectId});
 90 	    embV.updateColors();
 91 	    */
 92     };
 93   }); // click listener
 94 
 95   heatmapOverlayArea.addEventListener('mousemove', function(e) {
 96 
 97     var aspHeatView =  new aspectHeatmapViewer();
 98 
 99     var x = e.layerX;
100     var y = e.layerY;
101 
102   	var heatV = new heatmapViewer();
103   	var metaV = new metaDataHeatmapViewer();
104 
105   	heatV.showOverlay(x);
106   	metaV.showOverlay(x);
107 
108   	aspHeatView.showOverlay(x, y);
109 
110   });
111 
112    heatmapOverlayArea.addEventListener('mouseenter', function(e) {
113      document.body.style.cursor = "crosshair";
114    });
115 
116    heatmapOverlayArea.addEventListener('mouseout', function(e) {
117      var aspHeatView =  new aspectHeatmapViewer();
118      aspHeatView.clearOverlay();
119      document.body.style.cursor = "default";
120    });
121 
122 
123 };

125 /**

128 aspectHeatmapViewer.prototype.clearOverlay = function() {
129   var overlayArea = document.getElementById('aspect-heatmap-area-overlay');
130   var ctx = overlayArea.getContext('2d');

136 /**
137  * Show overlay for specific coordinates
138  */
139 aspectHeatmapViewer.prototype.showOverlay = function(x,y) {
140   var aspHeatView = new aspectHeatmapViewer()
141 
142   var overlayArea = document.getElementById('aspect-heatmap-area-overlay');
143   var ctx = overlayArea.getContext('2d');
144 
145   var drawConsts = aspHeatView.getDrawConstants();
146 
147   var areaWidth = overlayArea.width;
148   var areaHeight = overlayArea.height;
149 
150   ctx.setLineDash([10,10])
151   ctx.lineWidth = 1;
152   ctx.clearRect(0,0,areaWidth,areaHeight);
153 
154   var actualPlotHeight = this.getActualPlotHeight();
155 
156   if (typeof y !== 'undefined' & y < actualPlotHeight & y > drawConsts.top){
157     ctx.beginPath();
158     ctx.moveTo(drawConsts.left, y);
159     ctx.lineTo(drawConsts.width + drawConsts.left, y);
160     ctx.stroke();
161   }
162 
163   if (typeof x !== 'undefined' & x > drawConsts.left & x < drawConsts.width + drawConsts.left  &
164 	    (y < actualPlotHeight  | typeof y === 'undefined') // if y is provided it is in the plot
165        ) {
166 
167 	ctx.beginPath();
168 	ctx.moveTo(x, drawConsts.top);
169 	ctx.lineTo(x, actualPlotHeight + drawConsts.top);
170 	ctx.stroke();
171   }
172 
173 }

193 
194   var aspectHeatmapArea = $('#aspect-heatmap-area')[0];
195   var aspectHeatmapAreaOverlay = $('#aspect-heatmap-area-overlay')[0];

199   var curWidth = heatDendV.getCurrentWidth();
200   var curHeight = heatDendV.getCurrentAspectHeatmapHeight();
201 
202   var containerDiv = $('#aspect-heatmap-container');
203   containerDiv.css({
204     'width': curWidth,
205     'height': curHeight
206   });
207 
208   this.canvasElementWidth = curWidth;
209   this.canvasElementHeight= curHeight;
210 
211   // Update the size of teh main area
212   aspectHeatmapArea.width = curWidth;
213   aspectHeatmapArea.height = curHeight;
214 
215   // Update the size of the overaly
216   aspectHeatmapAreaOverlay.width = curWidth;
217   aspectHeatmapAreaOverlay.height = curHeight;
218 }

227 /**
228  * Draw the heatmap. Will clear heatmap if required
229  */
230 aspectHeatmapViewer.prototype.drawHeatmap = function() {
231 	var aspHeatView = this;
232   var heatDendView = new heatmapDendrogramViewer();
233 	var dendV = new dendrogramViewer();
234 	ctx = this.getDrawingContext();
235 
236 	drawConsts = this.getDrawConstants();
237 	var top = drawConsts.top;
238 	var left = drawConsts.left;
239 	var heatmapWidth  = drawConsts.width;
240 	var heatmapHeight = drawConsts.height - drawConsts.paddingBottom;
241 
242 	// Get the cells to plot
243 	var cellRange = dendV.getCurrentDisplayCellsIndexes();
244 	var cellIndexStart = cellRange[0];
245 	var cellIndexEnd = cellRange[1];
246 
247 	this.clearHeatmap(ctx);
248 
249 
250 	// Show centered waiting icon
251 	$('#aspect-heatmap-container').append("<img class='loadingIcon' src='img/loading.gif'/>");
252 	var loadingDomItem =  $('#aspect-heatmap-container > .loadingIcon');
253   var lpad = heatDendView.getCurrentWidth()  / 2;
254   var tpad = heatDendView.getCurrentHeatmapHeight() /2;
255   loadingDomItem.css({'padding-left': lpad + 'px', 'padding-top': tpad + 'px'});
256 
257 	var dataCntr = new dataController();
258 	dataCntr.getAspectMatrix(cellIndexStart, cellIndexEnd, false, function(data){
259 
260 	  loadingDomItem.remove();
261 
262     var naspects = data.Dim[1];
263     var ncells = data.Dim[0];
264 
265     // Computed plotting params
266     var cellWidth = heatmapWidth / ncells;
267     var cellHeight = heatmapHeight / naspects;
268 
269     var actualPlotHeight = heatmapHeight; // for convinience
270     aspHeatView.setActualPlotHeight( actualPlotHeight );
271 
272     var palSize = aspHeatView.palManager.getNumberOfColors();
273     var pal = aspHeatView.palManager.getPaletteColors();
274 
275 
276 	  ctx.fillStyle = pal[Math.floor(palSize/2)]
277 	  ctx.fillRect(left, top, heatmapWidth, actualPlotHeight);
278 
279     for (var j = 0; j < data.p.length -1; j++) {
280       var rsi = data.p[j];
281       var rei = data.p[j+1] - 1;
282 
283       if (rsi === rei) {continue;};
284 
285       // Calculate row normalisation
286       var rowMin = data.x.slice(rsi, rei).reduce( function(a,b){ return Math.min(a,b)} );
287 	    var rowMax = data.x.slice(rsi, rei).reduce(function(a,b){ return Math.max(a,b) } );
288 	    var rowSum = data.x.slice(rsi, rei).reduce(function(a,b){ return a+b });
289       var rowMean = rowSum / (rei - rsi + 1);
290 	    var maxAbsValue = Math.max(Math.abs(rowMin - rowMean), Math.abs(rowMax - rowMean));
291 
292       // Color mapper for this row
293       var colorMapper = aspHeatView.palManager.getMeanClampedColorMapper(rowMean, maxAbsValue, palSize);
294 
295       for (var k = rsi; k < rei; k++){
296         var palIndex = colorMapper(data.x[k]);
297         ctx.fillStyle = pal[palIndex];
298 
299         var x = data.i[k] * cellWidth + left;
300 		    var y = j * cellHeight + top;
301 
302 		    ctx.fillRect(x,y, cellWidth, cellHeight);
303 	    } // for k
304 
305 
306     } // for j
307 
308     // Plot the bounding box
309     ctx.strokeRect(left, top, heatmapWidth, actualPlotHeight);
310 
311 
312     // TODO: setup click areas here
313     aspHeatView.aspectRegions.clearClickAreas();
314     for (var i = 0; i < data.DimNames2.length; i++){
315       var x1 = left;
316 	    var y1 = cellHeight * i + top;
317 	    var x2 = heatmapWidth;
318 	    var y2 = cellHeight * ( i + 1 ) + top;
319 
320 	    aspHeatView.aspectRegions.addClickArea(
321     		x1, y1,
322     		x1, y2,
323     		x2, y2,
324     		x2, y1,
325     		{aspectId: data.DimNames2[i] }
326 	    );
327     }
328 
329 
330 	}); // dataController callback
331 
332 };
