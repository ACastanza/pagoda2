 20 	    for (var j = 0; j < row.length; j++) {
 21 		max =  Math.max(max, row[j]);

 25 
 26 	return [min, max];

102      * @param a {Array[]} The first array
103      * @param b {Array[]} The second array
104      */

106 	if (a.length != b.length) return false;
107 
108 	ac = a.slice();
109 	bc = b.slice();
110 
111 	as = ac.sort();
112 	bs = bc.sort();
113 
114 	for (var i = 0; i < as.length; i++) {
115 	    if (as[i] !== bs[i]) return false;
116 	}
117 
118 	return true;

155     /**
156      * Perform hierarchical clustring in a way similar to R
157      * The code is translated from https://github.com/bwlewis/hclust_in_R/blob/master/hc.R
158      * @param d a full distance matrix obtained  somehow, square
159      */
160     hclust: function(d) {
161 	// Assume that d is square
162 	var N = d.length;
163 
164 	// Set diagonal to infinity
165 	for (var i = 0; i < N; i++) {
166 	    d[i][i] = Infinity;
167 	}
168 
169 	// Tracks group membership
170 	var n = math.multiply(pagHelpers.seq(1, N), -1);
171 
172 	// The merge output N-1 x 2
173 	var m = [];
174 	for (var i = 0; i < N-1; i++) {
175 	    m[i] = new Array(2);
176 	}
177 
178 	// The height output
179 	var h = pagHelpers.rep(0, N-1);
180 
181 	// The merge steps
182 	var s = pagHelpers.seq(0, N-2);
183 	for (let j of s ) {
184 	    // Find the smallest distance and between 
185 	    // which elements it occurs
186 
187 	    // returns distance, index1, index2
188 	    minElement = pagHelpers.minElementAndPosition2d(d);
189 	    
190 	    // Save the distance in the height for this step
191 	    h[j] = minElement[0];
192 
193 	    // The groups in which the two merges belong
194 	    p = [ Number(n[minElement[1]]), Number(n[minElement[2]]) ];
195 
196 	    // Sort so if we are merging an element and a group 
197 	    // the group will be second
198 	    p.sort();
199 
200 	    // Save the merge step
201 	    m[j] = p;
202 	   
203 	    // Get the current membership of the elements we are merging
204 	    var elem1membership = n[minElement[1]];
205 	    var elem2membership = n[minElement[2]];
206 
207 	    // Set all elements that share this membership to the current cluster
208 	    for (var y = 0; y < n.length; y++) {
209 		if (n[y] === elem1membership || n[y] === elem2membership) {
210 		    n[y] = j
211 		}
212 	    };
213 
214 	    // Set membership of the elements we are merging
215 	    n[minElement[1]] = j;
216 	    n[minElement[2]] = j;
217 
218 
219 	    // Update distances 
220 
221 	    // This is a vector of the updated memberships
222 	    // Get the mean of the distances ignoring infinities
223 	    var r = [] ; // Our new distances
224 	    d1 = d[minElement[1]]
225 	    d2 = d[minElement[2]]
226 	    for (var l = 0; l < d1.length; l++) {
227 		if (d1[l] === Infinity) {
228 		    r[l] = d2[l];
229 		} else if (d2[l] === Infinity) {
230 		    r[l] = d1[l];
231 		} else {
232 		    r[l] = (d1[l] + d2[l] ) /2;
233 		}
234 	    }
235 
236 	    // Modify the distance matrix
237 
238 	    min_i = Math.min(minElement[1], minElement[2]);
239 	    // Set row of minElement[1]
240 	    d[min_i] = r;
241 	    // Set col of minElement[1]
242 	    for (var l = 0; l < d.length; l++) {
243 		d[l][min_i] = r[l];
244 	    }
245 
246 	    d[min_i][min_i] = Infinity;
247 
248 	    max_i = Math.max(minElement[1], minElement[2]);
249 
250 	    d[max_i]  = pagHelpers.rep(Infinity, N);
251 	    for (var l = 0; l < d.length; l++) {
252 		d[l][max_i] = pagHelpers.rep(Infinity, N);
253 	    }
254 	} // for ( j in s )
255 
256 	return {
257 	    merge: m,
258 	    height: h,
259 	    order: pagHelpers.iorder(m)
260 	};
261 
262     },

264     /** 
265      * Obtain an ordering for a given merge matrix
266      * this is going to be slow...
267      */
268     iorder: function(m) {
269 	// The number of elements (merges + 1)
270 	var N = m.length + 1;
271 
272 	// Start with the last merge in the
273 	// first two positions
274 	var iorder = pagHelpers.rep(0,N);
275 	iorder[0] = m[N-2][0];
276 	iorder[1] = m[N-2][1];
277 
278 	// This is the position in the iorder array up to which we are using
279 	var loc = 1;
280 	
281 	// For all the merges (except one) from the end backwards
282 	var i_seq = pagHelpers.seq(N - 3, 0, -1);
283 
284 	for ( var i in i_seq ) {
285 	    i = i_seq[i];
286 	
287 	    var j_seq = pagHelpers.seq(0, loc);
288 	    for ( var j in j_seq) {
289 		j = j_seq[j];
290 
291 		if (iorder[j] === i) {
292 		    iorder[j] = m[i][0];
293 		    
294 		    if (j === loc) {
295 			// At the end of the inner loop
296 			loc = loc + 1;
297 			iorder[loc] = m[i][1];
298 		    } else {
299 
300 			loc = loc + 1;
301 			k_seq = pagHelpers.seq(loc,j+1,-1);
302 			for( var k in k_seq ) { k = k_seq[k]; iorder[k] = iorder[k-1];}
303 			iorder[j+1] = m[i][1];
304 		    }
305 
306 		} // if (iorder[j] === i) {
307 
308 	    }
309 	};
310 	
311 	//return iorder;
312 
313 	// Return order of elements
314 	return math.subtract(math.multiply(iorder, -1),1);
315     },

317     /**
318      * Find the minimum element position in a 2d array
319      * @param array a 2d array
320      * @return an array [minimumValue,rowIndex, colIndex]
321      */
322     minElementAndPosition2d: function(array) {
323 	var minRow, minCol;
324 
325 	var nrow = array.length;
326 	var ncol = array[0].length;
327 	
328 	var curMin = Infinity;
329 
330 	for (var i = 0; i < nrow; i++) {
331 	    for (var j = 0; j < ncol; j++) {
332 		if (curMin > array[i][j]) {
333 		    curMin = array[i][j];
334 		    minRow = i;
335 		    minCol = j;
336 		}
337 	    }
338 	};
339 
340 	return [curMin, minRow, minCol];
341     },

408     /**
409      * Calculate the correlation distance of rows of a 2d array
410      */
411     matrixRowCorrDist: function(array) {
412 	var dist = [];
413 
414 	// set jj to 0
415 	for ( var i = 0; i < array.length; i++) {
416 	    dist[i] = [];
417 	    for ( var j = 0; j <= i; j++) {
418 		if (i === j) {
419 		    dist[i][j] = 0;
420 		} else {
421 		    dist[i][j] = 1 - math.abs(pagHelpers.correlation(array[i], array[j]));
422 		    dist[j][i] = dist[i][j];
423 		}
424 	    }
425 	}
426 
427 	return dist;
428     },

430     /**
431      * Calculate the pearson correlation between a and b.
432      * a and b are two 1d arrays of equal length
433      */ 
434     correlation: function(a, b) {
435 	// NOTE: using math.js
436 	
437 	a_mean = math.mean(a);
438 	b_mean = math.mean(b);
439 
440 	a_std = math.std(a, 'uncorrected');
441 	b_std = math.std(b, 'uncorrected');
442 
443 	// cov(x,y) = E[(x - mu_x)(y - my_y)]
444 	var covariance = math.mean(
445 	    math.dotMultiply(math.subtract(a, a_mean),
446 			  math.subtract(b, b_mean))
447 	);
448 
449 	// rho = cov(x,y) / ( rho_x * rho_y 
450 	var correlation = covariance / (a_std * b_std);
451 
452 	return correlation;
453     },

455     /**
456      * Generate a sequence of integers
457      * @param start
458      * @param end
459      * @param step
460      * @return array of sequence
461      */
462     seq: function(start, end, step = 1) {
463 	var r = [];
464 	
465 	if ( step > 0) {
466 	    var j =0;
467 	    for (var i = start; i <= end; i = i + step){
468 		r[j] = i;
469 		j++;
470 	    };
471 	} else if ( step < 0 ) {
472 	    var j = 0;
473 	    for (var i = start; i >= end; i = i + step) {
474 		r[j] = i;
475 		j++;
476 	    };
477 	} else {
478 	    throw new Error('Step can not be 0');
479 	}
480 
481 	return r;
482     }
